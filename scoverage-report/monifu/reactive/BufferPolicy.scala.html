17 package monifu.reactive
18 
19 /**
20  * Represents the buffering policy chosen for actions that need buffering,
21  * instructing the pipeline what to do when the buffer is full.
22  *
23  * For the available policies, see:
24  *
25  * - [[BufferPolicy.Unbounded Unbounded]]
26  * - [[BufferPolicy.OverflowTriggering OverflowTriggering]]
27  * - [[BufferPolicy.BackPressured BackPressured]]
28  *
29  * Used in [[monifu.reactive.observers.BufferedSubscriber BufferedSubscriber]]
30  * to implement buffering when concurrent actions are needed, such as in
31  * [[monifu.reactive.Channel Channels]] or in [[monifu.reactive.Observable.merge Observable.merge]].
32  */
33 sealed trait BufferPolicy[+T]
34 
35 object BufferPolicy {
36   /**
37    * A category of [[BufferPolicy]] for buffers that can be used
38    * synchronously, without worrying about back-pressure concerns.
39    *
40    * Needed such that buffer policies can safely be used
41    * in combination with [[Channel]] for publishing. For now,
42    * that's all policies except [[BackPressured]], a policy
43    * that can't work for [[Channel]].
44    */
45   sealed trait Synchronous[+T] extends BufferPolicy[T]
46   
47   /**
48    * A [[BufferPolicy]] specifying that the buffer is completely unbounded.
49    * Using this policy implies that with a fast data source, the system's
50    * memory can be exhausted and the process might blow up on lack of memory.
51    */
52   case object Unbounded extends Synchronous[Nothing]
53 
54   /**
55    * A [[BufferPolicy]] specifying that on reaching the maximum size,
56    * the pipeline should cancel the subscription and send an `onError`
57    * to the observer(s) downstream.
58    *
59    * @param bufferSize specifies how many events our buffer can hold
60    *                   before overflowing
61    */
62   case class OverflowTriggering(bufferSize: Int)
63     extends Synchronous[Nothing] {
64 
65     require(bufferSize > 1, "bufferSize must be strictly greater than 1")
66   }
67 
68   /**
69    * A [[BufferPolicy]] specifying that on reaching the maximum size,
70    * the pipeline should try to apply back-pressure (i.e. it should try
71    * delaying the data source in producing more elements, until the
72    * the consumer has drained the buffer and space is available).
73    *
74    * @param bufferSize specifies how many events our buffer can hold
75    *                   before overflowing
76    */
77   case class BackPressured(bufferSize: Int)
78     extends BufferPolicy[Nothing] {
79 
80     require(bufferSize > 1, "bufferSize should be strictly greater than 1")
81   }
82 
83   /**
84    * A [[BufferPolicy]] specifying that on reaching the maximum size,
85    * the pipeline should begin dropping incoming events until the buffer
86    * has room in it again and is free to process more elements.
87    *
88    * @param bufferSize specifies how many events our buffer can hold
89    *                   before overflowing
90    */
91   case class DropIncoming[T](bufferSize: Int)
92     extends Synchronous[Nothing] {
93 
94     require(bufferSize > 1, "bufferSize should be strictly greater than 1")
95   }
96 
97   /**
98    * A [[BufferPolicy]] specifying that on reaching the maximum size,
99    * the pipeline should begin dropping incoming events until the buffer
100    * has room in it again and is free to process more elements.
101    *
102    * @param bufferSize specifies how many events our buffer can hold
103    *                   before overflowing
104    * @param onOverflow is a function that receives the number of events that were
105    *                   dropped and is used to construct an overflow event for
106    *                   signalling downstream that events were dropped.
107    */
108   case class DropIncomingThenSignal[+T](bufferSize: Int, onOverflow: Long => T)
109     extends Synchronous[T] {
110         
111     require(bufferSize > 1, "bufferSize should be strictly greater than 1")
112   }
113 
114   /**
115    * The default library-wide policy used whenever a default argument
116    * value is needed.
117    */
118   val default: BufferPolicy[Nothing] = BackPressured(bufferSize = 2048)
119 }
