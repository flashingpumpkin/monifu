25 /**
26  * A `BehaviorChannel` is a [[Channel]] that uses an underlying
27  * [[monifu.reactive.subjects.BehaviorSubject BehaviorSubject]].
28  */
29 final class BehaviorChannel[T] private (initialValue: T, policy: BufferPolicy.Synchronous[T])
30     (implicit val scheduler: Scheduler)
31   extends Channel[T] with Observable[T] {
32 
33   private[this] val lock = new AnyRef
34   private[this] val subject = BehaviorSubject(initialValue)
35   private[this] val channel = BufferedSubscriber(subject, policy)
36 
37   private[this] var isDone = false
38   private[this] var lastValue = initialValue
39   private[this] var errorThrown = null : Throwable
40 
41   def subscribeFn(subscriber: Subscriber[T]): Unit = {
42     subject.unsafeSubscribe(subscriber)
43   }
44 
45   def pushNext(elems: T*): Unit = lock.synchronized {
46     if (!isDone)
47       for (elem <- elems) {
48         lastValue = elem
49         channel.observer.onNext(elem)
50       }
51   }
52 
53   def pushComplete() = lock.synchronized {
54     if (!isDone) {
55       isDone = true
56       channel.observer.onComplete()
57     }
58   }
59 
60   def pushError(ex: Throwable) = lock.synchronized {
61     if (!isDone) {
62       isDone = true
63       errorThrown = ex
64       channel.observer.onError(ex)
65     }
66   }
67 
68   def :=(update: T): Unit = pushNext(update)
69 
70   def apply(): T = lock.synchronized {
71     if (errorThrown ne null)
72       throw errorThrown
73     else
74       lastValue
75   }
76 }
