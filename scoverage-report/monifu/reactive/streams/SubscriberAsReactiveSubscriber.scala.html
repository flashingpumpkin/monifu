23 /**
24  * Wraps a [[monifu.reactive.Observer Observer]] instance into an
25  * `org.reactivestreams.Subscriber` instance. The resulting
26  * subscriber respects the [[http://www.reactive-streams.org/ Reactive Streams]]
27  * contract.
28  *
29  * Given that when emitting [[monifu.reactive.Observer.onNext Observer.onNext]] calls,
30  * the call may pass asynchronous boundaries, the emitted events need to be buffered.
31  * The `requestCount` constructor parameter also represents the buffer size.
32  * 
33  * To create an instance, [[SubscriberAsReactiveSubscriber.apply]] must be used: {{{
34  *   // uses the default requestCount of 128
35  *   val subscriber = SubscriberAsReactiveSubscriber(new Observer[Int] {
36  *     private[this] var sum = 0
37  *     
38  *     def onNext(elem: Int) = {
39  *       sum += elem
40  *       Continue
41  *     }
42  *     
43  *     def onError(ex: Throwable) = {
44  *       logger.error(ex)
45  *     }
46  *     
47  *     def onComplete() = {
48  *       logger.info("Stream completed")
49  *     }
50  *   })
51  * }}}
52  *
53  * @param subscriber the observer instance that will get wrapped into a
54  *                   `org.reactivestreams.Subscriber`, along with the scheduler used
55  *
56  * @param requestCount the parameter passed to `Subscription.request`,
57  *                    also representing the buffer size; MUST BE strictly positive
58  */
59 final class SubscriberAsReactiveSubscriber[T] private
60     (subscriber: monifu.reactive.Subscriber[T], requestCount: Int)
61   extends Subscriber[T] {
62 
63   require(requestCount > 0, "requestCount must be strictly positive, according to the Reactive Streams contract")
64 
65   private[this] val buffer =
66     SynchronousSubscriberAsReactiveSubscriber(
67       SynchronousBufferedSubscriber.unbounded(subscriber.observer)(subscriber.scheduler),
68       requestCount = requestCount
69     )
70 
71   def onSubscribe(s: Subscription): Unit =
72     buffer.onSubscribe(s)
73 
74   def onNext(elem: T): Unit =
75     buffer.onNext(elem)
76 
77   def onError(ex: Throwable): Unit =
78     buffer.onError(ex)
79 
80   def onComplete(): Unit =
81     buffer.onComplete()
82 }

87    * Wraps a [[monifu.reactive.Observer Observer]] instance into a
88    * `org.reactivestreams.Subscriber` instance. The resulting
89    * subscriber respects the [[http://www.reactive-streams.org/ Reactive Streams]]
90    * contract.
91    *
92    * Given that when emitting [[monifu.reactive.Observer.onNext Observer.onNext]] calls,
93    * the call may pass asynchronous boundaries, the emitted events need to be buffered.
94    * The `requestCount` constructor parameter also represents the buffer size.
95    *
96    * To create an instance, [[SubscriberAsReactiveSubscriber.apply]] must be used: {{{
97    *   // uses the default requestCount of 128
98    *   val subscriber = SubscriberAsReactiveSubscriber(new Observer[Int] {
99    *     private[this] var sum = 0
100    *
101    *     def onNext(elem: Int) = {
102    *       sum += elem
103    *       Continue
104    *     }
105    *
106    *     def onError(ex: Throwable) = {
107    *       logger.error(ex)
108    *     }
109    *
110    *     def onComplete() = {
111    *       logger.info("Stream completed")
112    *     }
113    *   })
114    * }}}
115    *
116    * @param subscriber the subscriber instance that will get wrapped into a
117    *                  `org.reactivestreams.Subscriber`
118    *
119    * @param requestCount the parameter passed to each `Subscription.request` call,
120    *                    also representing the buffer size; MUST BE strictly positive
121    */
122   def apply[T](subscriber: monifu.reactive.Subscriber[T], requestCount: Int = 128): Subscriber[T] =
123     subscriber match {
124       case ref: SynchronousSubscriber[_] =>
125         SynchronousSubscriberAsReactiveSubscriber(ref.asInstanceOf[SynchronousSubscriber[T]], requestCount)
126       case _ =>
127         new SubscriberAsReactiveSubscriber[T](subscriber, requestCount)
128     }
129 }

131 /**
132  * Wraps a [[monifu.reactive.observers.SynchronousObserver SynchronousObserver]] instance into a
133  * `org.reactivestreams.Subscriber` instance. The resulting
134  * subscriber respects the [[http://www.reactive-streams.org/ Reactive Streams]]
135  * contract.
136  *
137  * Given that we can guarantee a [[monifu.reactive.observers.SynchronousObserver SynchronousObserver]]
138  * is used, then no buffering is needed and thus the implementation is very efficient.
139  *
140  * To create an instance, [[SynchronousSubscriberAsReactiveSubscriber.apply]] must be used: {{{
141  *   // uses the default requestCount of 128
142  *   val subscriber = SynchronousSubscriberAsReactiveSubscriber(new Observer[Int] {
143  *     private[this] var sum = 0
144  *
145  *     def onNext(elem: Int) = {
146  *       sum += elem
147  *       Continue
148  *     }
149  *
150  *     def onError(ex: Throwable) = {
151  *       logger.error(ex)
152  *     }
153  *
154  *     def onComplete() = {
155  *       logger.info("Stream completed")
156  *     }
157  *   })
158  * }}}
159  */
160 final class SynchronousSubscriberAsReactiveSubscriber[T] private
161     (subscriber: SynchronousSubscriber[T], requestCount: Int)
162   extends Subscriber[T] {
163 
164   require(requestCount > 0, "requestCount must be strictly positive, according to the Reactive Streams contract")
165 
166   private[this] implicit val s = subscriber.scheduler
167 
168   private[this] var subscription = null : Subscription
169   private[this] var expectingCount = 0
170   @volatile private[this] var isCanceled = false
171 
172   def onSubscribe(s: Subscription): Unit =
173     if (!isCanceled) {
174       subscription = s
175       expectingCount = requestCount
176       s.request(requestCount)
177     }
178 
179   def onNext(elem: T): Unit = {
180     if (!isCanceled) {
181       if (expectingCount > 0) expectingCount -= 1
182 
183       subscriber.observer.onNext(elem) match {
184         case Continue =>
185           // should it request more events?
186           if (expectingCount == 0) {
187             expectingCount = requestCount
188             subscription.request(requestCount)
189           }
190         case Cancel =>
191           // downstream canceled, so we MUST cancel too
192           isCanceled = true
193           subscription.cancel()
194       }
195     }
196   }
197 
198   def onError(ex: Throwable): Unit = {
199     if (!isCanceled) {
200       isCanceled = true
201       subscriber.observer.onError(ex)
202     }
203   }
204 
205   def onComplete(): Unit = {
206     if (!isCanceled) {
207       isCanceled = true
208       subscriber.observer.onComplete()
209     }
210   }
211 }
