29 /**
30  * Wraps a `org.reactivestreams.Subscriber` instance that respects the
31  * [[http://www.reactive-streams.org/ Reactive Streams]] contract
32  * into an [[monifu.reactive.Observer Observer]] instance that respect the `Observer`
33  * contract.
34  */
35 final class SubscriberAsObserver[T] private
36     (subscriber: RSubscriber[T])(implicit s: Scheduler)
37   extends Observer[T] {
38 
39   import monifu.reactive.streams.SubscriberAsObserver.RequestsQueue
40 
41   private[this] val requests = new RequestsQueue
42   private[this] var leftToPush = 0L
43   private[this] var firstEvent = true
44 
45   @tailrec
46   def onNext(elem: T): Future[Ack] = {
47     if (firstEvent) {
48       firstEvent = false
49       subscriber.onSubscribe(createSubscription())
50       onNext(elem) // retry
51     }
52     else if (leftToPush > 0) {
53       leftToPush -= 1
54       subscriber.onNext(elem)
55       Continue
56     }
57     else {
58       requests.await().flatMap { requested =>
59         if (requested <= 0) Cancel else {
60           leftToPush += (requested - 1)
61           subscriber.onNext(elem)
62           Continue
63         }
64       }
65     }
66   }
67 
68   def onError(ex: Throwable): Unit = {
69     subscriber.onError(ex)
70   }
71 
72   def onComplete(): Unit = {
73     subscriber.onComplete()
74   }
75 
76   private def createSubscription() = new Subscription {
77     def cancel(): Unit = {
78       requests.cancel()
79     }
80 
81     def request(n: Long): Unit = {
82       require(n > 0, "n must be strictly positive, according to " +
83         "the Reactive Streams contract, rule 3.9")
84       requests.request(n)
85     }
86   }
87 }

89 object SubscriberAsObserver {
90   /**
91    * Given an `org.reactivestreams.Subscriber` as defined by
92    * the [[http://www.reactive-streams.org/ Reactive Streams]]
93    * specification, it builds an [[Observer]] instance compliant
94    * with the Monifu Rx implementation.
95    */
96   def apply[T](subscriber: RSubscriber[T])(implicit s: Scheduler): SubscriberAsObserver[T] = {
97     new SubscriberAsObserver[T](subscriber)
98   }
99 
100   /**
101    * An asynchronous queue implementation for dealing with
102    * requests from a Subscriber.
103    */
104   private final class RequestsQueue {
105     private[this] val state = Atomic(ActiveState(Queue.empty, Queue.empty) : State)
106 
107     @tailrec
108     def await(): Future[Long] = {
109       state.get match {
110         case CancelledState =>
111           Future.successful(0)
112 
113         case oldState @ ActiveState(elements, promises) =>
114           if (elements.nonEmpty) {
115             val (e, newQ) = elements.dequeue
116             val newState = ActiveState(newQ, promises)
117 
118             if (!state.compareAndSet(oldState, newState))
119               await()
120             else
121               Future.successful(e)
122           }
123           else {
124             val p = Promise[Long]()
125             val newState = ActiveState(elements, promises.enqueue(p))
126 
127             if (!state.compareAndSet(oldState, newState))
128               await()
129             else
130               p.future
131           }
132       }
133     }
134 
135     @tailrec
136     def request(n: Long): Unit = {
137       require(n > 0, "n must be strictly positive, according to " +
138         "the Reactive Streams contract, rule 3.9")
139 
140       state.get match {
141         case CancelledState =>
142           () // do nothing
143 
144         case oldState @ ActiveState(elements, promises) if promises.nonEmpty =>
145           val (p, q) = promises.dequeue
146           val newState = ActiveState(elements, q)
147 
148           if (!state.compareAndSet(oldState, newState))
149             request(n)
150           else
151             p.success(n)
152 
153         case oldState @ ActiveState(Queue(requested), promises) if requested > 0 =>
154           val r = requested + n
155           val newState = ActiveState(Queue(r), promises)
156 
157           if (!state.compareAndSet(oldState, newState))
158             request(n)
159 
160         case oldState @ ActiveState(elements, promises) =>
161           val newState = ActiveState(elements.enqueue(n), promises)
162           if (!state.compareAndSet(oldState, newState))
163             request(n)
164       }
165     }
166 
167     @tailrec
168     def cancel(): Unit = {
169       state.get match {
170         case CancelledState =>
171           () // do nothing
172 
173         case oldState @ ActiveState(_, promises) =>
174           if (!state.compareAndSet(oldState, CancelledState))
175             cancel()
176           else
177             promises.foreach(_.success(0))
178       }
179     }
180 
181     sealed trait State
182 
183     case class ActiveState(elements: Queue[Long], promises: Queue[Promise[Long]])
184       extends State
185 
186     case object CancelledState
187       extends State
188   }
