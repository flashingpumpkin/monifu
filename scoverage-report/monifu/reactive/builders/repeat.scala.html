22 object repeat {
23   /**
24    * Creates an Observable that continuously emits the given ''item'' repeatedly.
25    */
26   def apply[T](elems: T*): Observable[T] = {
27     if (elems.size == 0)
28       Observable.empty
29     else if (elems.size == 1) {
30       Observable.create { subscriber =>
31         implicit val s = subscriber.scheduler
32         val o = subscriber.observer
33 
34         def loop(elem: T): Unit =
35           o.onNext(elem).onSuccess {
36             case Continue =>
37               loop(elem)
38           }
39 
40         loop(elems.head)
41       }
42     }
43     else
44       Observable.fromIterable(elems).repeat
45   }
46 }
