25 object interval {
26   /**
27    * Creates an Observable that emits auto-incremented natural numbers (longs) spaced equally by
28    * a given time interval. Starts from 0 with no delay, after which it emits incremented
29    * numbers spaced by the `period` of time.
30    *
31    * 
32    *
33    * @param delay the delay between two subsequent events
34    */
35   def withFixedDelay(delay: FiniteDuration): Observable[Long] = {
36     Observable.create { subscriber =>
37       implicit val s = subscriber.scheduler
38       val o = subscriber.observer
39 
40       s.execute(new Runnable { self =>
41         import monifu.reactive.internals.ObserverState.{ON_CONTINUE, ON_NEXT}
42         var state = ON_NEXT
43         var counter = 0L
44 
45         def run() = state match {
46           case ON_NEXT =>
47             state = ON_CONTINUE
48             o.onNext(counter).onContinue(self)
49 
50           case ON_CONTINUE =>
51             state = ON_NEXT
52             counter += 1
53             s.scheduleOnce(delay, self)
54         }
55       })
56     }
57   }
58 
59   /**
60    * Creates an Observable that emits auto-incremented natural numbers (longs) emitted
61    * at a fixed rate, as specified by `period`.
62    *
63    * @param period the delay between two subsequent events
64    */
65   def atFixedRate(period: FiniteDuration): Observable[Long] = {
66     Observable.create { subscriber =>
67       implicit val s = subscriber.scheduler
68       val o = subscriber.observer
69 
70       s.execute(new Runnable { self =>
71         import monifu.reactive.internals.ObserverState.{ON_CONTINUE, ON_NEXT}
72 
73         private[this] val periodMillis = period.toMillis
74         private[this] var state = ON_NEXT
75         private[this] var counter = 0L
76         private[this] var startedAt = 0L
77 
78         def run() = state match {
79           case ON_NEXT =>
80             state = ON_CONTINUE
81             startedAt = s.currentTimeMillis()
82             o.onNext(counter).onContinue(self)
83 
84           case ON_CONTINUE =>
85             state = ON_NEXT
86             counter += 1
87 
88             val delay = {
89               val durationMillis = s.currentTimeMillis() - startedAt
90               val d = periodMillis - durationMillis
91               if (d >= 0L) d else 0L
92             }
93 
94             s.scheduleOnce(delay, TimeUnit.MILLISECONDS, self)
95         }
96       })
97     }
98   }
