30     Observable.create[Notification[T]] { subscriber =>
31       implicit val s = subscriber.scheduler
32       val observer = subscriber.observer
33 
34       source.unsafeSubscribe(new Observer[T] {
35         def onNext(elem: T): Future[Ack] = {
36           observer.onNext(OnNext(elem))
37         }
38 
39         def onError(ex: Throwable): Unit = {
40           observer.onNext(OnError(ex))

42         }
43

45           observer.onNext(OnComplete)

47         }
48       })
49     }
