25 object repeat {
26   /**
27    * Implementation for [[Observable.repeat]].
28    */
29   def apply[T](source: Observable[T]): Observable[T] = {
30     // recursive function - subscribes the observer again when
31     // onComplete happens
32     def loop(subject: Subject[T, T], observer: Observer[T])(implicit s: Scheduler): Unit =
33       subject.unsafeSubscribe(new Observer[T] {
34         def onNext(elem: T) = {
35           observer.onNext(elem)
36         }
37 
38         def onError(ex: Throwable) =
39           observer.onError(ex)
40 
41         def onComplete(): Unit =
42           loop(subject, observer)
43       })
44 
45     Observable.create { subscriber =>
46       implicit val s = subscriber.scheduler
47       val observer = subscriber.observer
48 
49       val subject = ReplaySubject[T]()
50       loop(subject, observer)
51 
52       source.unsafeSubscribe(new Observer[T] {
53         def onNext(elem: T): Future[Ack] = {
54           subject.onNext(elem)
55         }
56 
57         def onError(ex: Throwable): Unit = {
58           subject.onError(ex)
59         }
60 
61         def onComplete(): Unit = {
62           subject.onComplete()
63         }
64       })
65     }
66   }
