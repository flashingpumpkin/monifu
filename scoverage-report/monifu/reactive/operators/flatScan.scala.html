29    * Implementation for [[Observable.flatScan]].
30    */
31   def apply[T,R](source: Observable[T], initial: R)(op: (R, T) => Observable[R]) =
32     Observable.create[R] { subscriber =>
33       implicit val s = subscriber.scheduler
34       val o = subscriber.observer
35 
36       source.unsafeSubscribe(new Observer[T] {
37         private[this] val refCount = RefCountCancelable(o.onComplete())
38         private[this] var state = initial
39 
40         def onNext(elem: T) = {
41           // for protecting user calls
42           var streamError = true
43           try {
44             val upstreamPromise = Promise[Ack]()
45             val newState = op(state, elem)
46             streamError = false
47 
48             val refID = refCount.acquire()
49 
50             newState.unsafeSubscribe(new Observer[R] {
51               def onNext(elem: R): Future[Ack] = {
52                 state = elem
53                 o.onNext(elem)
54                   .ifCancelTryCanceling(upstreamPromise)
55               }
56 
57               def onError(ex: Throwable): Unit = {
58                 // error happened, so signaling both the main thread that it should stop
59                 // and the downstream consumer of the error
60                 upstreamPromise.trySuccess(Cancel)
61                 o.onError(ex)
62               }
63 
64               def onComplete(): Unit = {
65                 // NOTE: we aren't sending this onComplete signal downstream to our observer
66                 // instead we are just instructing upstream to send the next observable
67                 upstreamPromise.trySuccess(Continue)
68                 refID.cancel()
69               }
70             })
71 
72             upstreamPromise.future
73           }
74           catch {
75             case NonFatal(ex) =>
76               if (streamError) {
77                 o.onError(ex)
78                 Cancel
79               }
80               else {
81                 Future.failed(ex)
82               }
83           }
84         }
85 
86         def onError(ex: Throwable) = {
87           // oops, error happened on main thread, piping that along should cancel everything
88           o.onError(ex)
89         }
90 
91         def onComplete() = {
92           refCount.cancel()
93         }
94       })
95     }
