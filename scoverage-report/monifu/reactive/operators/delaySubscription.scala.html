25 object delaySubscription {
26   /**
27    * Implementation for [[Observable.delaySubscription]].
28    */
29   def onFuture[T](source: Observable[T], future: Future[_]): Observable[T] =
30       Observable.create[T] { subscriber =>
31         implicit val s = subscriber.scheduler
32         val observer = subscriber.observer
33 
34         future.onComplete {
35           case Success(_) =>
36             source.unsafeSubscribe(observer)
37           case Failure(ex) =>
38             observer.onError(ex)
39         }
40       }
41 
42   /**
43    * Implementation for [[Observable.delaySubscription]].
44    */
45   def onTimespan[T](source: Observable[T], timespan: FiniteDuration): Observable[T] =
46     Observable.create { subscriber =>
47       implicit val s = subscriber.scheduler
48       val underlying = source.delaySubscription {
49         val p = Promise[Unit]()
50         s.scheduleOnce(timespan)(p.success(()))
51         p.future
52       }
53 
54       underlying.unsafeSubscribe(subscriber)
55     }
