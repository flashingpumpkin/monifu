27 object whileBusy {
28   /**
29    * While the destination observer is busy, drop the incoming events.
30    */
31   def dropEvents[T](source: Observable[T]): Observable[T] =
32     Observable.create { subscriber =>
33       implicit val s = subscriber.scheduler
34       val observer = subscriber.observer
35 
36       source.unsafeSubscribe(new SynchronousObserver[T] {
37         private[this] var lastAck = Continue : Future[Ack]
38         private[this] var isDone = false
39 
40         def onNext(elem: T) = {
41           if (!isDone) lastAck match {
42             case sync if sync.isCompleted =>
43               sync.value.get match {
44                 case Success(Cancel) =>
45                   isDone = true
46                   Cancel
47 
48                 case Failure(ex) =>
49                   isDone = true
50                   observer.onError(ex)
51                   Cancel
52 
53                 case Success(Continue) =>
54                   observer.onNext(elem) match {
55                     case Cancel =>
56                       isDone = true
57                       Cancel
58                     case other =>
59                       lastAck = other
60                       Continue
61                   }
62               }
63 
64             case _ =>
65               Continue
66           }
67           else
68             Cancel
69         }
70 
71         def onError(ex: Throwable) =
72           if (!isDone) {
73             isDone = true
74             lastAck.onContinueSignalError(observer, ex)
75           }
76 
77         def onComplete() =
78           if (!isDone) {
79             isDone = true
80             lastAck.onContinueSignalComplete(observer)
81           }
82       })
83     }

85   /**
86    * While the destination observer is busy,
87    * drop the incoming events, then signal how many events
88    * where dropped.
89    */
90   def dropEventsThenSignalOverflow[T](source: Observable[T], onOverflow: Long => T): Observable[T] =
91     Observable.create { subscriber =>
92       implicit val s = subscriber.scheduler
93       val observer = subscriber.observer
94 
95       source.unsafeSubscribe(new SynchronousObserver[T] {
96         private[this] var lastAck = Continue : Future[Ack]
97         private[this] var eventsDropped = 0L
98         private[this] var isDone = false
99 
100         def onNext(elem: T) = {
101           if (!isDone) lastAck match {
102             case sync if sync.isCompleted =>
103               sync.value.get match {
104                 case Success(Cancel) =>
105                   isDone = true
106                   Cancel
107 
108                 case Failure(ex) =>
109                   isDone = true
110                   observer.onError(ex)
111                   Cancel
112 
113                 case Success(Continue) =>
114                   val hasOverflow = eventsDropped > 0
115                   var streamError = true
116 
117                   lastAck = if (hasOverflow)
118                     try {
119                       val message = onOverflow(eventsDropped)
120                       eventsDropped = 0
121                       streamError = false
122                       observer.onNext(message)
123                     }
124                     catch {
125                       case NonFatal(ex) if streamError =>
126                         onError(ex)
127                         Cancel
128                     }
129                   else {
130                     observer.onNext(elem)
131                   }
132 
133                   if (hasOverflow)
134                     onNext(elem) // retry
135                   else
136                     Continue
137               }
138 
139             case _ =>
140               eventsDropped += 1
141               Continue
142           }
143           else
144             Cancel
145         }
146 
147         def onError(ex: Throwable): Unit = if (!isDone) {
148           isDone = true
149           lastAck.onContinueSignalError(observer, ex)
150         }
151 
152         def onComplete(): Unit = if (!isDone) {
153           isDone = true
154 
155           val f = if (eventsDropped <= 0) lastAck else {
156             var streamError = true
157             try {
158               val message = onOverflow(eventsDropped)
159               eventsDropped = 0
160               streamError = false
161 
162               lastAck.fastFlatMap {
163                 case Continue => observer.onNext(message)
164                 case Cancel => Cancel
165               }
166             }
167             catch {
168               case NonFatal(ex) =>
169                 Future.failed(ex)
170             }
171           }
172 
173           f.onContinueSignalComplete(observer)
174         }
175       })
176     }

178   /**
179    * Implementation for [[Observable.whileBusyBufferEvents]].
180    */
181   def bufferEvents[T](source: Observable[T], bufferSize: Int): Observable[Seq[T]] =
182     Observable.create { subscriber =>
183       implicit val s = subscriber.scheduler
184       val buffer = WhileBusyBufferSubscriber(subscriber.observer, bufferSize)
185       val downstream = buffer.observer
186 
187       source.unsafeSubscribe(new Observer[T] {
188         def onNext(elem: T) = {
189           downstream.onNext(elem)
190         }
191 
192         def onError(ex: Throwable) = {
193           downstream.onError(ex)
194         }
195 
196         def onComplete() = {
197           downstream.onComplete()
198         }
199       })
200     }
