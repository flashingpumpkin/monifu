27    * Implementation for [[monifu.reactive.Observable.reduce]].
28    */
29   def apply[T](source: Observable[T])(op: (T, T) => T): Observable[T] = {
30     Observable.create { subscriber =>
31       implicit val s = subscriber.scheduler
32       val observer = subscriber.observer
33 
34       source.unsafeSubscribe(new Observer[T] {
35         private[this] var state: T = _
36         private[this] var isFirst = true
37         private[this] var wasApplied = false
38 
39         def onNext(elem: T): Future[Ack] = {
40           // See Section 6.4. in the Rx Design Guidelines:
41           // Protect calls to user code from within an operator
42           try {
43             if (isFirst) {
44               isFirst = false
45               state = elem
46             }
47             else {
48               state = op(state, elem)
49               if (!wasApplied) wasApplied = true
50             }
51 
52             Continue
53           }
54           catch {
55             case NonFatal(ex) =>
56               onError(ex)
57               Cancel
58           }
59         }
60 
61         def onComplete() = {
62           if (wasApplied) {
63             observer.onNext(state)
64               .onContinueSignalComplete(observer)
65           }
66           else {
67             observer.onComplete()
68           }
69         }
70 
71         def onError(ex: Throwable) = {
72           observer.onError(ex)
73         }
74       })
75     }
76   }
