29 object debounce {
30   /**
31    * Only emit an item from an Observable if a particular 
32    * timespan has passed without it emitting another item
33    */
34   def apply[T](source: Observable[T], timeout: FiniteDuration): Observable[T] = {
35     Observable.create { subscriber => 
36       implicit val s = subscriber.scheduler
37       val downstream = subscriber.observer
38       val timeoutMillis = timeout.toMillis
39 
40       source.unsafeSubscribe(new SynchronousObserver[T] with Runnable { self =>
41         private[this] val task = MultiAssignmentCancelable()
42         private[this] var ack: Future[Ack] = Continue
43         private[this] var isDone = false
44         private[this] var lastEvent: T = _
45         private[this] var lastTSInMillis: Long = 0L
46 
47         locally {
48           scheduleNext(timeoutMillis)
49         }
50 
51         def scheduleNext(delayMillis: Long): Unit = {
52           task := s.scheduleOnce(delayMillis, TimeUnit.MILLISECONDS, self)
53         }
54 
55         def run(): Unit = self.synchronized {
56           if (!isDone) {
57             if (lastEvent == null) scheduleNext(timeoutMillis) else {
58               val rightNow = s.currentTimeMillis()
59               val sinceLastOnNext = rightNow - lastTSInMillis
60 
61               if (sinceLastOnNext >= timeoutMillis) {
62                 ack = downstream.onNext(lastEvent).fastFlatMap {
63                   case Continue =>
64                     val executionTime = s.currentTimeMillis() - rightNow
65                     val delay = if (timeoutMillis > executionTime)
66                       timeoutMillis - executionTime else 0L
67 
68                     scheduleNext(delay)
69                     Continue
70 
71                   case Cancel =>
72                     self.synchronized { isDone = true }
73                     Cancel
74                 }
75               }
76               else {
77                 val remainingTime = timeoutMillis - sinceLastOnNext
78                 scheduleNext(remainingTime)
79               }
80             }
81           }
82         }
83 
84         def onNext(elem: T): Ack = self.synchronized {
85           if (!isDone) {
86             lastEvent = elem
87             lastTSInMillis = s.currentTimeMillis()
88             Continue
89           }
90           else {
91             Cancel
92           }
93         }
94         
95         def onError(ex: Throwable): Unit = 
96           self.synchronized {
97             if (!isDone) {
98               isDone = true
99               task.cancel()
100               ack.onContinueSignalError(downstream, ex)
101               ack = Cancel
102             }
103           }
104 
105         def onComplete(): Unit = 
106           self.synchronized {
107             if (!isDone) {
108               isDone = true
109               task.cancel()
110               ack.onContinueSignalComplete(downstream)
111               ack = Cancel
112             }
113           }
114       })
115     }
116   }
117 }
