25   /**
26    * Implementation for [[Observable.onErrorRecoverWith]].
27    */
28   def recoverWith[T](source: Observable[T], pf: PartialFunction[Throwable, Observable[T]]) =
29     Observable.create[T] { subscriber =>
30       implicit val s = subscriber.scheduler
31       val o = subscriber.observer
32 
33       source.unsafeSubscribe(new Observer[T] {
34         def onNext(elem: T) =
35           o.onNext(elem)
36 
37         def onError(ex: Throwable) = {
38           // protecting user level code
39           var streamError = true
40           try {
41             if (pf.isDefinedAt(ex)) {
42               val fallbackTo = pf(ex)
43               // need asynchronous execution to avoid a synchronous loop
44               // blowing out the call stack
45               s.execute(fallbackTo.unsafeSubscribe(o))
46             }
47             else {
48               // we can't protect the onError call and if it throws
49               // the behavior should be undefined
50               streamError = false
51               o.onError(ex)
52             }
53           }
54           catch {
55             case NonFatal(err) if streamError =>
56               // streaming the immediate exception
57               try o.onError(err) finally {
58                 // logging the original exception
59                 s.reportFailure(ex)
60               }
61           }
62         }
63 
64         def onComplete() =
65           o.onComplete()
66       })
67     }

103   /**
104    * Implementation for [[Observable.onErrorRetry]].
105    */
106   def retryCounted[T](source: Observable[T], maxRetries: Long) = {
107     // helper to subscribe in a loop when onError happens
108     def subscribe(o: Observer[T], retryIdx: Long)(implicit s: Scheduler): Unit =
109       source.unsafeSubscribe(new Observer[T] {
110         def onNext(elem: T) = o.onNext(elem)
111         def onComplete() = o.onComplete()
112 
113         def onError(ex: Throwable) = {
114           if (retryIdx < maxRetries) {
115             // need asynchronous execution to avoid a synchronous loop
116             // blowing out the call stack
117             s.execute(subscribe(o, retryIdx+1))
118           }
119           else {
120             o.onError(ex)
121           }
122         }
123       })
124 
125     Observable.create[T] { s =>
126       subscribe(s.observer, 0)(s.scheduler)
127     }
128   }

152   /**
153    * Implementation for [[Observable.onErrorRetryIf]].
154    */
155   def retryIf[T](source: Observable[T], p: Throwable => Boolean) = {
156     // helper to subscribe in a loop when onError happens
157     def subscribe(o: Observer[T])(implicit s: Scheduler): Unit =
158       source.unsafeSubscribe(new Observer[T] {
159         def onNext(elem: T) = o.onNext(elem)
160         def onComplete() = o.onComplete()
161 
162         def onError(ex: Throwable) = {
163           // protecting against user level code
164           try {
165             val shouldRetry = p(ex)
166             // need asynchronous execution to avoid a synchronous loop
167             // blowing out the call stack
168             if (shouldRetry)
169               s.execute(subscribe(o))
170             else
171               o.onError(ex)
172           }
173           catch {
174             case NonFatal(err) =>
175               // exception is getting lost, so try logging it
176               s.reportFailure(ex)
177               // reporting user code exception, which
178               // is always worse if it happens
179               o.onError(err)
180           }
181         }
182       })
183 
184     Observable.create[T] { s =>
185       subscribe(s.observer)(s.scheduler)
186     }
187   }
