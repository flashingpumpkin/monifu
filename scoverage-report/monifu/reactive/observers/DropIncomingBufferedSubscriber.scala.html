29 /**
30  * A [[BufferedSubscriber]] implementation for the following policies:
31  *
32  * - [[monifu.reactive.BufferPolicy.DropIncoming]]
33  * - [[monifu.reactive.BufferPolicy.DropIncomingThenSignal]]
34  */
35 final class DropIncomingBufferedSubscriber[-T] private
36     (underlying: Observer[T], bufferSize: Int, onOverflow: Long => T = null)
37     (implicit val scheduler: Scheduler)
38   extends BufferedSubscriber[T] with SynchronousSubscriber[T] {
39 
40   require(bufferSize > 0, "bufferSize must be a strictly positive number")
41 
42   private[this] val queue = ConcurrentQueue.empty[T]
43   // to be modified only in onError, before upstreamIsComplete
44   private[this] var errorThrown: Throwable = null
45   // to be modified only in onError / onComplete
46   @volatile private[this] var upstreamIsComplete = false
47   // to be modified only by consumer
48   @volatile private[this] var downstreamIsDone = false
49   // for enforcing non-concurrent updates
50   private[this] val itemsToPush = Atomic(0)
51 
52   private[this] var eventsDropped = 0L
53 
54   val observer: SynchronousObserver[T] = new SynchronousObserver[T] {
55     def onNext(elem: T): Ack = {
56       if (!upstreamIsComplete && !downstreamIsDone) {
57         if (itemsToPush.get >= bufferSize) {
58           // no more room, dropping event
59           eventsDropped += 1
60           Continue
61         }
62         else if (eventsDropped > 0 && onOverflow != null) {
63           try {
64             val message = onOverflow(eventsDropped)
65             eventsDropped = 0
66             // first send the overflow message
67             queue.offer(message)
68             notifyConsumerOfNewEvent()
69             // then try to send our current event (recursive call)
70             onNext(elem)
71           }
72           catch {
73             case NonFatal(ex) =>
74               onError(ex)
75               Cancel
76           }
77         }
78         else try {
79           queue.offer(elem)
80           notifyConsumerOfNewEvent()
81           Continue
82         }
83         catch {
84           case NonFatal(ex) =>
85             onError(ex)
86             Cancel
87         }
88       }
89       else
90         Cancel
91     }
92 
93     def onError(ex: Throwable) = {
94       if (!upstreamIsComplete && !downstreamIsDone) {
95         errorThrown = ex
96         upstreamIsComplete = true
97         notifyConsumerOfNewEvent()
98       }
99     }
100 
101     def onComplete(): Unit = {
102       if (!upstreamIsComplete && !downstreamIsDone) {
103         if (eventsDropped > 0 && onOverflow != null) try {
104           val message = onOverflow(eventsDropped)
105           eventsDropped = 0
106           queue.offer(message)
107           upstreamIsComplete = true
108           notifyConsumerOfNewEvent()
109         }
110         catch {
111           case NonFatal(ex) =>
112             onError(ex)
113         }
114         else {
115           upstreamIsComplete = true
116           notifyConsumerOfNewEvent()
117         }
118       }
119     }
120   }
121 
122   @tailrec
123   private[this] def notifyConsumerOfNewEvent(): Unit = {
124     val currentNr = itemsToPush.get
125 
126     if (!itemsToPush.compareAndSet(currentNr, currentNr + 1))
127       notifyConsumerOfNewEvent()
128     else if (currentNr == 0)
129       scheduler.execute(new Runnable {
130         def run() = fastLoop(0)
131       })
132   }
133 
134   private[this] def rescheduled(processed: Int): Unit = {
135     fastLoop(processed)
136   }
137 
138   @tailrec
139   private[this] def fastLoop(processed: Int): Unit = {
140     if (!downstreamIsDone) {
141       val hasError = errorThrown ne null
142       val next = queue.poll()
143 
144       if (next != null) {
145         underlying.onNext(next) match {
146           case sync if sync.isCompleted =>
147             sync match {
148               case continue if continue == Continue || continue.value.get == Continue.IsSuccess =>
149                 // process next
150                 fastLoop(processed + 1)
151 
152               case done if done == Cancel || done.value.get == Cancel.IsSuccess =>
153                 // ending loop
154                 downstreamIsDone = true
155                 itemsToPush.set(0)
156 
157               case error if error.value.get.isFailure =>
158                 // ending loop
159                 downstreamIsDone = true
160                 itemsToPush.set(0)
161                 underlying.onError(error.value.get.failed.get)
162             }
163 
164           case async =>
165             async.onComplete {
166               case Continue.IsSuccess =>
167                 // re-run loop (in different thread)
168                 rescheduled(processed + 1)
169 
170               case Cancel.IsSuccess =>
171                 // ending loop
172                 downstreamIsDone = true
173                 itemsToPush.set(0)
174 
175               case Failure(ex) =>
176                 // ending loop
177                 downstreamIsDone = true
178                 itemsToPush.set(0)
179                 underlying.onError(ex)
180 
181               case other =>
182                 // never happens, but to appease Scala's compiler
183                 downstreamIsDone = true
184                 itemsToPush.set(0)
185                 underlying.onError(new MatchError(s"$other"))
186             }
187         }
188       }
189       else if (upstreamIsComplete || hasError) {
190         // Race-condition check, but if upstreamIsComplete=true is visible,
191         // then the queue should be fully published because there's a clear happens-before
192         // relationship between queue.offer() and upstreamIsComplete=true
193         if (!queue.isEmpty) {
194           fastLoop(processed)
195         }
196         else {
197           // ending loop
198           downstreamIsDone = true
199           itemsToPush.set(0)
200           queue.clear() // for GC purposes
201           if (errorThrown ne null)
202             underlying.onError(errorThrown)
203           else
204             underlying.onComplete()
205         }
206       }
207       else {
208         val remaining = itemsToPush.decrementAndGet(processed)
209         // if the queue is non-empty (i.e. concurrent modifications just happened)
210         // then start all over again
211         if (remaining > 0) fastLoop(0)
212       }
213     }
214   }
215 }
