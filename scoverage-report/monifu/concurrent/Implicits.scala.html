19 import monifu.concurrent.schedulers.AsyncScheduler

21 import scala.concurrent.ExecutionContext
22 
23 /**
24  * Defines implicit values that can be used by importing in the 
25  * current context.
26  * 
27  * Example:
28  * {{{
29  *   import monifu.concurrent.Implicits.globalScheduler
30  * }}}
31  */
32 object Implicits {
33   /**
34    * A global [[Scheduler]] instance, provided for convenience, piggy-backing
35    * on top of Scala's own `concurrent.ExecutionContext.global`, which is a
36    * `ForkJoinPool`.
37    *
38    * It can be tuned by setting the following JVM system properties:
39    *
40    * - "scala.concurrent.context.minThreads" an integer specifying the minimum
41    *   number of active threads in the pool
42    *
43    * - "scala.concurrent.context.maxThreads" an integer specifying the maximum
44    *   number of active threads in the pool
45    *
46    * - "scala.concurrent.context.numThreads" can be either an integer,
47    *   specifying the parallelism directly or a string with the format "xNUM"
48    *   (e.g. "x1.5") specifying the multiplication factor of the number of
49    *   available processors (taken with `Runtime.availableProcessors`)
50    *
51    * The formula for calculating the parallelism in our pool is
52    * `min(maxThreads, max(minThreads, numThreads))`.
53    *
54    * To set a system property from the command line, a JVM parameter must be
55    * given to the `java` command as `-Dname=value`. So as an example, to customize
56    * this global scheduler, we could start our process like this:
57    *
58    * 
59    *   java -Dscala.concurrent.context.minThreads=2 \
60    *        -Dscala.concurrent.context.maxThreads=30 \
61    *        -Dscala.concurrent.context.numThreads=x1.5 \
62    *        ...
63    * 

64    *
65    * As a note, this being backed by Scala's own global execution context,
66    * it is cooperating with Scala's BlockContext, so when operations marked
67    * with `scala.concurrent.blocking` are encountered, the thread-pool may
68    * decide to add extra threads in the pool. However this is not a thread-pool
69    * that is optimal for doing blocking operations, so for example if you want
70    * to do a lot of blocking I/O, then use a Scheduler backed by a
71    * thread-pool that is more optimal for blocking. See for example
72    * [[Scheduler.io]].
73    */
74   implicit lazy val globalScheduler: Scheduler =
75     AsyncScheduler(
76       Scheduler.defaultScheduledExecutor,
77       ExecutionContext.Implicits.global,
78       UncaughtExceptionReporter.LogExceptionsToStandardErr
79     )
80 }
