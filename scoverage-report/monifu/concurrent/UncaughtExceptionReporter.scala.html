17 package monifu.concurrent
18 
19 import scala.annotation.implicitNotFound
20 import scala.concurrent.ExecutionContext
21 
22 /**
23  * An exception reporter is a function that logs an uncaught error.
24  *
25  * Usually taken as an implicit when executing computations that could fail,
26  * but that must not blow up the call-stack, like asynchronous tasks.
27  *
28  * A default implicit is provided that simply logs the error on STDERR.
29  */
30 @implicitNotFound(
31   "No ExceptionReporter was found in context for " +
32   "reporting uncaught errors, either build one yourself or use either " +
33   "an implicit Scheduler (schedulers are ExceptionReporters)")
34 trait UncaughtExceptionReporter {
35   def reportFailure(ex: Throwable): Unit
36 }
37 
38 /**
39  * See [[UncaughtExceptionReporter]].
40  */
41 object UncaughtExceptionReporter {
42   /**
43    * Builds a reporter out of the provided callback.
44    */
45   def apply(reporter: Throwable => Unit): UncaughtExceptionReporter =
46     new UncaughtExceptionReporter {
47       def reportFailure(ex: Throwable) = reporter(ex)
48     }
49 
50   /**
51    * The default reporter.
52    * Simply prints stack traces on STDERR.
53    */
54   val LogExceptionsToStandardErr = {
55     UncaughtExceptionReporter(ex => ExecutionContext.defaultReporter(ex))
56   }
57 }
