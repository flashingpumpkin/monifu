77   /**
78    * Creates a [[Scheduler]] meant for computational heavy tasks.
79    *
80    * Characteristics:
81    *
82    * - backed by Scala's `ForkJoinPool` for the task execution, in async mode
83    * - uses Monifu's default `ScheduledExecutorService` instance for scheduling
84    * - all created threads are daemonic
85    * - cooperates with Scala's `BlockContext`
86    *
87    * @param parallelism is the number of threads that can run in parallel
88    * @param r is the [[UncaughtExceptionReporter]] that logs uncaught exceptions.
89    */
90   def computation(parallelism: Int, r: UncaughtExceptionReporter = LogExceptionsToStandardErr): Scheduler = {
91     val exceptionHandler = new UncaughtExceptionHandler {
92       def uncaughtException(t: Thread, e: Throwable) =
93         r.reportFailure(e)
94     }
95 
96     val pool = new scala.concurrent.forkjoin.ForkJoinPool(
97       parallelism,
98       ForkJoinPool.defaultForkJoinWorkerThreadFactory,
99       exceptionHandler,
100       true // asyncMode
101     )
102 
103     val context = ExecutionContext.fromExecutor(pool, r.reportFailure)
104     AsyncScheduler(defaultScheduledExecutor, context, r)
105   }

107   /**
108    * Creates a [[Scheduler]] meant for blocking I/O tasks.
109    *
110    * Characteristics:
111    *
112    * - backed by a cached `ThreadPool` executor with 60 seconds of keep-alive
113    * - the maximum number of threads is unbounded, as recommended for blocking I/O
114    * - uses Monifu's default `ScheduledExecutorService` instance for scheduling
115    * - doesn't cooperate with Scala's `BlockContext` only because it is unbounded
116    *
117    * @param name the created threads name prefix, for easy identification.
118    *
119    * @param daemonic specifies whether the created threads should be daemonic
120    *                 (non-daemonic threads are blocking the JVM process on exit).
121    *
122    * @param r is the [[UncaughtExceptionReporter]] that logs uncaught exceptions.
123    */
124   def io(name: String = "monifu-io", daemonic: Boolean = true,
125     r: UncaughtExceptionReporter = LogExceptionsToStandardErr): Scheduler = {
126     val threadFactory = new ThreadFactory {
127       private[this] val counter = Atomic(0L)
128       def newThread(r: Runnable): Thread = {
129         val th = new Thread(r)
130         th.setDaemon(daemonic)
131         th.setName(name + "-" + counter.getAndIncrement().toString)
132         th
133       }
134     }
135 
136     val context = ExecutionContext.fromExecutor(
137       Executors.newCachedThreadPool(threadFactory),
138       r.reportFailure
139     )
140 
141     AsyncScheduler(defaultScheduledExecutor, context, r)
142   }

144   /**
145    * Builds a [[Scheduler]] that schedules and executes tasks on its own thread.
146    *
147    * Characteristics:
148    *
149    * - backed by a single-threaded `ScheduledExecutorService` that takes care
150    *   of both scheduling tasks in the future and of executing tasks
151    * - does not cooperate with Scala's `BlockingContext`, so tasks should not
152    *   block on the result of other tasks scheduled to run on this same thread
153    *
154    * @param name is the name of the created thread, for easy identification
155    *
156    * @param daemonic specifies whether the created thread should be daemonic
157    *                 (non-daemonic threads are blocking the JVM process on exit)
158    *
159    * @param r is the [[UncaughtExceptionReporter]] that logs uncaught exceptions.
160    */
161   def singleThread(name: String, daemonic: Boolean = true,
162     r: UncaughtExceptionReporter = LogExceptionsToStandardErr): Scheduler = {
163     val executor =
164       Executors.newSingleThreadScheduledExecutor(new ThreadFactory {
165         def newThread(r: Runnable) = {
166           val th = new Thread(r)
167           th.setName(name)
168           th.setDaemon(daemonic)
169           th
170         }
171       })
172 
173     val context = new ExecutionContext {
174       def reportFailure(t: Throwable) = r.reportFailure(t)
175       def execute(runnable: Runnable) = executor.execute(runnable)
176     }
177 
178     AsyncScheduler(executor, context, r)
179   }
180 
181   /**
182    * The default `ScheduledExecutor` instance.
183    */
184   private[concurrent] lazy val defaultScheduledExecutor =
185     Executors.newSingleThreadScheduledExecutor(new ThreadFactory {
186       def newThread(r: Runnable): Thread = {
187         val th = new Thread(r)
188         th.setDaemon(true)
189         th.setName("monifu-scheduler")
190         th
191       }
192     })
