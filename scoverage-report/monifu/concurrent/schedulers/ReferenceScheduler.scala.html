83   def scheduleAtFixedRate(initialDelay: FiniteDuration, period: FiniteDuration, r: Runnable): Cancelable = {
84     val sub = MultiAssignmentCancelable()
85 
86     def loop(initialDelay: FiniteDuration): Unit = {
87       val startedAtMillis = currentTimeMillis()
88 
89       sub := scheduleOnce(initialDelay, new Runnable {
90         def run(): Unit = {
91           r.run()
92 
93           val delay = {
94             val duration = (currentTimeMillis() - startedAtMillis).millis
95             val d = period - duration

102     }

113       val startedAtMillis = currentTimeMillis()
114 
115       sub := scheduleOnce(initialDelayMs, TimeUnit.MILLISECONDS, new Runnable {
116         def run(): Unit = {
117           r.run()
118 
119           val delay = {
120             val durationMillis = currentTimeMillis() - startedAtMillis

123           }
124 
125           loop(delay, periodMs)
126         }
127       })
128     }
129
