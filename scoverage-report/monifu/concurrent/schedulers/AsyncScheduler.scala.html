25 /**
26  * An `AsyncScheduler` schedules tasks to happen in the future with the
27  * given `ScheduledExecutorService` and the tasks themselves are executed on
28  * the given `ExecutionContext`.
29  */
30 final class AsyncScheduler private
31     (s: ScheduledExecutorService, ec: ExecutionContext, r: UncaughtExceptionReporter)
32   extends ReferenceScheduler {
33 
34   def scheduleOnce(initialDelay: FiniteDuration, r: Runnable): Cancelable = {
35     scheduleOnce(initialDelay.length, initialDelay.unit, r)
36   }
37 
38   def scheduleOnce(initialDelay: Long, unit: TimeUnit, r: Runnable) = {
39     if (initialDelay <= 0) {
40       execute(r)
41       Cancelable()
42     }
43     else {
44       val task = s.schedule(r, initialDelay, unit)
45       Cancelable(task.cancel(true))
46     }
47   }
48 
49   override def scheduleWithFixedDelay(initialDelay: FiniteDuration, delay: FiniteDuration, r: Runnable): Cancelable = {
50     val task = s.scheduleWithFixedDelay(r, initialDelay.toMillis, delay.toMillis, TimeUnit.MILLISECONDS)
51     Cancelable(task.cancel(false))
52   }
53 
54   override def scheduleWithFixedDelay(initialDelay: Long, delay: Long, unit: TimeUnit, r: Runnable): Cancelable = {
55     val task = s.scheduleWithFixedDelay(r, initialDelay, delay, unit)
56     Cancelable(task.cancel(false))
57   }
58 
59   override def scheduleAtFixedRate(initialDelay: FiniteDuration, period: FiniteDuration, r: Runnable): Cancelable = {
60     val task = s.scheduleAtFixedRate(r, initialDelay.toMillis, period.toMillis, TimeUnit.MILLISECONDS)
61     Cancelable(task.cancel(false))
62   }
63 
64   override def scheduleAtFixedRate(initialDelay: Long, period: Long, unit: TimeUnit, r: Runnable): Cancelable = {
65     val task = s.scheduleAtFixedRate(r, initialDelay, period, unit)
66     Cancelable(task.cancel(false))
67   }
68 
69   def execute(runnable: Runnable): Unit =
70     ec.execute(runnable)
71 
72   def reportFailure(t: Throwable): Unit =
73     r.reportFailure(t)
74 }
75 
76 object AsyncScheduler {
77   def apply(schedulerService: ScheduledExecutorService,
78             ec: ExecutionContext, reporter: UncaughtExceptionReporter): AsyncScheduler =
79     new AsyncScheduler(schedulerService, ec, reporter)
80 }
