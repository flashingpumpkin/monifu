264     val current = get
265     if (current != 0) {
266       val decrement = if (current >= v) v else current
267       val update = minusOp(current, decrement)
268       if (!compareAndSet(current, update))
269         countDownToZero(v)
270       else
271         decrement
272     }
273     else
274       0
275   }

283   private[this] final def plusOp(a: Char, b: Char): Char =
284     ((a + b) & mask).asInstanceOf[Char]
285 
286   private[this] final def minusOp(a: Char, b: Char): Char =
287     ((a - b) & mask).asInstanceOf[Char]
288 
289   private[this] final def incrOp(a: Char, b: Int): Char =
290     ((a + b) & mask).asInstanceOf[Char]
