17 package monifu.concurrent.atomic
18 
19 /**
20  * Base trait of all atomic references, no matter the type.
21  */
22 trait Atomic[T] extends Any {
23   /**
24    * @return the current value persisted by this Atomic
25    */
26   def get: T
27 
28   /**
29    * @return the current value persisted by this Atomic, an alias for `get()`
30    */
31   def apply(): T = get
32 
33   /**
34    * Updates the current value.
35    * @param update will be the new value returned by `get()`
36    */
37   def set(update: T): Unit
38 
39   /**
40    * Alias for `set()`. Updates the current value.
41    * @param value will be the new value returned by `get()`
42    */
43   def update(value: T): Unit
44 
45   /**
46    * Alias for `set()`. Updates the current value.
47    * @param value will be the new value returned by `get()`
48    */
49   def `:=`(value: T): Unit
50 
51   /**
52    * Does a compare-and-set operation on the current value. For more info, checkout the related
53    * [[https://en.wikipedia.org/wiki/Compare-and-swap Compare-and-swap Wikipedia page]].
54    *
55    * It's an atomic, worry free operation.
56    *
57    * @param expect is the value you expect to be persisted when the operation happens
58    * @param update will be the new value, should the check for `expect` succeeds
59    * @return either true in case the operation succeeded or false otherwise
60    */
61   def compareAndSet(expect: T, update: T): Boolean
62 
63   /**
64    * Sets the persisted value to `update` and returns the old value that was in place.
65    * It's an atomic, worry free operation.
66    */
67   def getAndSet(update: T): T
68 
69   /**
70    * Eventually sets to the given value. Has weaker visibility guarantees than the normal `set()`.
71    *
72    */
73   def lazySet(update: T): Unit
74 
75   /**
76    * Abstracts over `compareAndSet`. You specify a transformation by specifying a callback to be
77    * executed, a callback that transforms the current value. This method will loop until it will
78    * succeed in replacing the current value with the one produced by your callback.
79    *
80    * Note that the callback will be executed on each iteration of the loop, so it can be called
81    * multiple times - don't do destructive I/O or operations that mutate global state in it.
82    *
83    * @param cb is a callback that receives the current value as input and returns a tuple that specifies
84    *           the update + what should this method return when the operation succeeds.
85    * @return whatever was specified by your callback, once the operation succeeds
86    */
87   def transformAndExtract[U](cb: (T) => (U, T)): U
88 
89   /**
90    * Abstracts over `compareAndSet`. You specify a transformation by specifying a callback to be
91    * executed, a callback that transforms the current value. This method will loop until it will
92    * succeed in replacing the current value with the one produced by the given callback.
93    *
94    * Note that the callback will be executed on each iteration of the loop, so it can be called
95    * multiple times - don't do destructive I/O or operations that mutate global state in it.
96    *
97    * @param cb is a callback that receives the current value as input and returns the `update` which is the
98    *           new value that should be persisted
99    * @return whatever the update is, after the operation succeeds
100    */
101   def transformAndGet(cb: (T) => T): T
102 
103   /**
104    * Abstracts over `compareAndSet`. You specify a transformation by specifying a callback to be
105    * executed, a callback that transforms the current value. This method will loop until it will
106    * succeed in replacing the current value with the one produced by the given callback.
107    *
108    * Note that the callback will be executed on each iteration of the loop, so it can be called
109    * multiple times - don't do destructive I/O or operations that mutate global state in it.
110    *
111    * @param cb is a callback that receives the current value as input and returns the `update` which is the
112    *           new value that should be persisted
113    * @return the old value, just prior to when the successful update happened
114    */
115   def getAndTransform(cb: (T) => T): T
116 
117   /**
118    * Abstracts over `compareAndSet`. You specify a transformation by specifying a callback to be
119    * executed, a callback that transforms the current value. This method will loop until it will
120    * succeed in replacing the current value with the one produced by the given callback.
121    *
122    * Note that the callback will be executed on each iteration of the loop, so it can be called
123    * multiple times - don't do destructive I/O or operations that mutate global state in it.
124    *
125    * @param cb is a callback that receives the current value as input and returns the `update` which is the
126    *           new value that should be persisted
127    */
128   def transform(cb: (T) => T): Unit
129 }
130 
131 object Atomic {
132   /**
133    * Constructs an `Atomic[T]` reference. Based on the `initialValue`, it will return the best, most specific
134    * type. E.g. you give it a number, it will return something inheriting from `AtomicNumber[T]`. That's why
135    * it takes an `AtomicBuilder[T, R]` as an implicit parameter - but worry not about such details as it just works.
136    *
137    * @param initialValue is the initial value with which to initialize the Atomic reference
138    * @param builder is the builder that helps us to build the best reference possible, based on our `initialValue`
139    */
140   def apply[T, R <: Atomic[T]](initialValue: T)(implicit builder: AtomicBuilder[T, R]): R =
141     builder.buildInstance(initialValue)
142 
143   /**
144    * Returns the builder that would be chosen to construct Atomic references
145    * for the given `initialValue`.
146    */
147   def builderFor[T, R <: Atomic[T]](initialValue: T)(implicit builder: AtomicBuilder[T, R]): AtomicBuilder[T, R] =
148     builder
149 }
