17 package monifu.concurrent.atomic
18 
19 /**
20  * Represents an Atomic reference holding a number, providing helpers for easily incrementing and decrementing it.
21  *
22  * @tparam T should be something that's Numeric
23  */
24 trait AtomicNumber[T] extends Atomic[T] {
25   def increment(v: Int = 1): Unit
26   def add(v: T): Unit
27   def decrement(v: Int = 1): Unit
28   def subtract(v: T): Unit
29 
30   def incrementAndGet(v: Int = 1): T
31   def addAndGet(v: T): T
32   def decrementAndGet(v: Int = 1): T
33   def subtractAndGet(v: T): T
34 
35   def getAndIncrement(v: Int = 1): T
36   def getAndAdd(v: T): T
37   def getAndDecrement(v: Int = 1): T
38   def getAndSubtract(v: T): T
39 
40   /**
41    * Decrements this number until it reaches zero.
42    *
43    * @return a number representing how much it was able to subtract, which
44    *         is a value between zero and `v`
45    */
46   def countDownToZero(v: T): T
47 
48   def `+=`(v: T): Unit
49   def `-=`(v: T): Unit
50 }
51 
52 object AtomicNumber {
53   def apply[T, R <: AtomicNumber[T]](initialValue: T)(implicit ev: Numeric[T], builder: AtomicBuilder[T, R]): R =
54     builder.buildInstance(initialValue)
55 }
