85 /**
86  * Provides a constructor for building cache-line padded
87  * [[monifu.concurrent.atomic.Atomic Atomic]] references.
88  */
89 object Atomic {
90   /**
91    * Constructs an `Atomic[T]` reference. Based on the `initialValue`, it will return the best, most specific
92    * type. E.g. you give it a number, it will return something inheriting from `AtomicNumber[T]`. That's why
93    * it takes an `AtomicBuilder[T, R]` as an implicit parameter - but worry not about such details as it just works.
94    *
95    * @param initialValue is the initial value with which to initialize the Atomic reference
96    * @param builder is the builder that helps us to build the best reference possible, based on our `initialValue`
97    */
98   def apply[T, R <: Atomic[T]](initialValue: T)(implicit builder: AtomicBuilder[T, R]): R =
99     builder.buildInstance(initialValue)
100 
101   /**
102    * Returns the builder that would be chosen to construct Atomic references
103    * for the given `initialValue`.
104    */
105   def builderFor[T, R <: Atomic[T]](initialValue: T)(implicit builder: AtomicBuilder[T, R]): AtomicBuilder[T, R] =
106     builder
107 }
