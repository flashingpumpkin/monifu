19 /**
20  * Atomic classes that are cache-padded for reducing cache contention,
21  * until JEP 142 and `@Contended` happens. See:
22  *
23  * http://mail.openjdk.java.net/pipermail/hotspot-dev/2012-November/007309.html
24  */
25 package object padded {
26   import monifu.concurrent.atomic

28   // defining useful type-aliases

40   type AtomicNumberAny[T] = atomic.AtomicNumberAny[T]
41   type BlockableAtomic[T] = atomic.BlockableAtomic[T]
42 
43   /**
44    * Constructs an `Atomic[T]` reference. Based on the `initialValue`, it will return the best, most specific
45    * type. E.g. you give it a number, it will return something inheriting from `AtomicNumber[T]`. That's why
46    * it takes an `AtomicBuilder[T, R]` as an implicit parameter - but worry not about such details as it just works.
47    *
48    * @param initialValue is the initial value with which to initialize the Atomic reference
49    * @param builder is the builder that helps us to build the best reference possible, based on our `initialValue`
50    */
51   def apply[T, R <: Atomic[T]](initialValue: T)(implicit builder: AtomicBuilder[T, R]): R =
52     builder.buildInstance(initialValue)
53 }
