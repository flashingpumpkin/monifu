17 package monifu.concurrent
18 
19 import scala.concurrent.TimeoutException
20 
21 /**
22  * A small toolkit of classes that support compare-and-swap semantics for safe mutation of variables.
23  *
24  * On top of the JVM, this means dealing with lock-free thread-safe programming. Also works on top of Javascript,
25  * with Scala.js (for good reasons, as Atomic references are still useful in non-multi-threaded environments).
26  *
27  * The backbone of Atomic references is this method:
28  * {{{
29  *   def compareAndSet(expect: T, update: T): Boolean
30  * }}}
31  *
32  * This method atomically sets a variable to the `update` value if it currently holds
33  * the `expect` value, reporting `true` on success or `false` on failure. The classes in this package
34  * also contain methods to get and unconditionally set values. They also support weak operations,
35  * defined in `WeakAtomic[T]`, such as (e.g. `weakCompareAndSet`, `lazySet`) or operations that
36  * block the current thread through ''spin-locking'', until a condition happens (e.g. `waitForCompareAndSet`),
37  * methods exposed by `BlockingAtomic[T]`.
38  *
39  * Building a reference is easy with the provided constructor, which will automatically return the
40  * most specific type needed (in the following sample, that's an `AtomicDouble`, inheriting from `AtomicNumber[T]`):
41  * {{{
42  *   val atomicNumber = Atomic(12.2)
43  *
44  *   atomicNumber.incrementAndGet()
45  *   // => 13.2
46  * }}}
47  *
48  * In comparison with `java.util.concurrent.AtomicReference`, these references implement common interfaces
49  * that you can use generically (i.e. `Atomic[T]`, `AtomicNumber[T]`, `BlockableAtomic[T]`, `WeakAtomic[T]`).
50  * And also provide useful helpers for atomically mutating of values
51  * (i.e. `transform`, `transformAndGet`, `getAndTransform`, etc...) or of numbers of any kind
52  * (`incrementAndGet`, `getAndAdd`, etc...).
53  *
54  * A high-level documentation describing the rationale for these can be found here:
55  * [[https://github.com/alexandru/monifu/blob/master/docs/atomic.md Atomic Reference]]
56  */
57 package object atomic {
58   /**
59    * For private use only by the `monifu` package.
60    *
61    * Checks if the current thread has been interrupted, throwing
62    * an `InterruptedException` in case it is.
63    */
64   @inline private[atomic] def interruptedCheck(): Unit = {
65     if (Thread.interrupted)
66       throw new InterruptedException()
67   }
68 
69   /**
70    * For private use only by the `monifu` package.
71    *
72    * Checks if the timeout is due, throwing a `TimeoutException` in case it is.
73    */
74   @inline private[atomic] def timeoutCheck(endsAtNanos: Long): Unit = {
75     if (System.nanoTime >= endsAtNanos)
76       throw new TimeoutException()
77   }
78 }
