21 /**
22  * Represents an asynchronous computation whose execution can be canceled.
23  * Used by [[monifu.concurrent.Scheduler]] giving you the ability to cancel scheduled units of work.
24  *
25  * It is equivalent to `java.io.Closeable`, but without the I/O focus, or to `IDisposable` in Microsoft .NET,
26  * or to `akka.actor.Cancellable`.
27  */
28 trait Cancelable {
29   /**
30    * Cancels the unit of work represented by this reference.
31    *
32    * Guaranteed idempotency - calling it multiple times should have the
33    * same side-effect as calling it only a single time. Implementations
34    * of this method should also be thread-safe.
35    *
36    * @return true if cancellation happened, or false if another execution
37    *         happened previously or concurrently.
38    */
39   def cancel(): Boolean
40 }
41 
42 object Cancelable {
43   def apply(callback: => Unit): Cancelable =
44     new Cancelable {
45       private[this] val _isCanceled = Atomic(false)
46 
47       def cancel(): Boolean = {
48         if (_isCanceled.compareAndSet(expect=false, update=true)) {
49           callback
50           true
51         }
52         else
53           false
54       }        
55     }
