25 /**
26  * Represents a [[monifu.concurrent.Cancelable]] whose underlying cancelable reference can be swapped for another.
27  *
28  * Example:
29  * {{{
30  *   val s = MultiAssignmentCancelable()
31  *   s() = c1 // sets the underlying cancelable to c1
32  *   s() = c2 // swaps the underlying cancelable to c2
33  *
34  *   s.cancel() // also cancels c2
35  *
36  *   s() = c3 // also cancels c3, because s is already canceled
37  * }}}
38  */
39 final class MultiAssignmentCancelable private () extends BooleanCancelable {
40   private[this] val state = Atomic(Active(Cancelable()) : State)
41 
42   def isCanceled: Boolean = state.get match {
43     case Cancelled => true
44     case _ => false
45   }
46 
47   @tailrec
48   def cancel(): Boolean = state.get match {
49     case Cancelled => false
50     case current @ Active(s) =>
51       if (state.compareAndSet(current, Cancelled)) {
52         s.cancel()
53         true
54       }
55       else
56         cancel()
57   }
58 
59   /**
60    * Swaps the underlying cancelable reference with `s`.
61    *
62    * In case this `MultiAssignmentCancelable` is already canceled,
63    * then the reference `value` will also be canceled on assignment.
64    */
65   @tailrec
66   def update(value: Cancelable): Unit = state.get match {
67     case Cancelled => value.cancel()
68     case current @ Active(s) =>
69       if (!state.compareAndSet(current, Active(value)))
70         update(value)
71   }
72 
73   /**
74    * Alias for `update(value)`
75    */
76   def `:=`(value: Cancelable): Unit =
77     update(value)
78 }
79 
80 object MultiAssignmentCancelable {
81   def apply(): MultiAssignmentCancelable =
82     new MultiAssignmentCancelable()
83 
84   def apply(s: Cancelable): MultiAssignmentCancelable = {
85     val ms = new MultiAssignmentCancelable()
86     ms() = s
87     ms
88   }
89 
90   sealed trait State
91   object State {
92     case class Active(s: Cancelable) extends State
93     case object Cancelled extends State
94   }
95 }
