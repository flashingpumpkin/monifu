19 import monifu.concurrent.atomic.AtomicAny
20 import scala.annotation.tailrec
21 import monifu.concurrent.Cancelable
22

24 /**
25  * Represents a [[monifu.concurrent.Cancelable]] that can be assigned only once to another
26  * cancelable reference.
27  *
28  * Similar to [[monifu.concurrent.cancelables.MultiAssignmentCancelable]], except that
29  * in case of multi-assignment, it throws a `java.lang.IllegalStateException`.
30  *
31  * If the assignment happens after this cancelable has been canceled, then on
32  * assignment the reference will get canceled too.
33  *
34  * Useful in case you need a forward reference.
35  */
36 final class SingleAssignmentCancelable private () extends BooleanCancelable {
37   import State._
38 
39   def isCanceled: Boolean = state.get match {
40     case IsEmptyCanceled | IsCanceled =>
41       true
42     case _ =>
43       false
44   }
45 
46   /**
47    * Sets the underlying cancelable reference with `s`.
48    *
49    * In case this `SingleAssignmentCancelable` is already canceled,
50    * then the reference `value` will also be canceled on assignment.
51    *
52    * @throws IllegalStateException in case this cancelable has already been assigned
53    */
54   @throws(classOf[IllegalStateException])
55   @tailrec
56   def update(value: Cancelable): Unit = state.get match {
57     case Empty =>
58       if (!state.compareAndSet(Empty, IsNotCanceled(value)))
59         update(value)
60     case IsEmptyCanceled =>
61       if (!state.compareAndSet(IsEmptyCanceled, IsCanceled))
62         update(value)
63       else
64         value.cancel()
65     case IsCanceled | IsNotCanceled(_) =>
66       throw new IllegalStateException("Cannot assign to SingleAssignmentCancelable, as it was already assigned once")
67   }
68 
69   @tailrec
70   def cancel(): Boolean = state.get match {
71     case Empty =>
72       if (!state.compareAndSet(Empty, IsEmptyCanceled))
73         cancel()
74       else
75         true
76     case old @ IsNotCanceled(s) =>
77       if (!state.compareAndSet(old, IsCanceled))
78         cancel()
79       else {
80         s.cancel()
81         true
82       }
83     case IsEmptyCanceled | IsCanceled =>
84       false
85   }
86 
87   /**
88    * Alias for `update(value)`
89    */
90   def `:=`(value: Cancelable): Unit =
91     update(value)
92 
93   private[this] val state = AtomicAny(Empty : State)
94 
95   private[this] sealed trait State
96   private[this] object State {
97     case object Empty extends State
98     case class IsNotCanceled(s: Cancelable) extends State
99     case object IsCanceled extends State
100     case object IsEmptyCanceled extends State
101   }
102 }
103 
104 object SingleAssignmentCancelable {
105   def  apply(): SingleAssignmentCancelable =
106     new SingleAssignmentCancelable()
107 }
