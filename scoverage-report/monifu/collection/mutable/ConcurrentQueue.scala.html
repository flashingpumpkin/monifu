26 /**
27  * An efficient queue data-structure.
28  *
29  * On top of the JVM it is backed by a `java.util.concurrent.ConcurrentLinkedQueue`
30  * so it can be used in a multi-threading context. On top of Scala.js / Javascript
31  * it is backed by a `scala.collection.mutable.Queue`.
32  *
33  * Contrary to Scala best-practices, the `poll()` method that pulls elements from our
34  * queue is returning `null` in case the queue is empty.
35  */
36 final class ConcurrentQueue[A](elems: A*)
37   extends Iterable[A]
38   with GenericTraversableTemplate[A, ConcurrentQueue]
39   with mutable.Cloneable[ConcurrentQueue[A]]
40   with Serializable {
41 
42   private[this] val underlying = {
43     val instance = new ConcurrentLinkedQueue[A]()
44     instance.addAll(elems.asJava)
45     instance
46   }
47 
48   override def companion: GenericCompanion[ConcurrentQueue] =
49     ConcurrentQueue
50 
51   override def newBuilder: mutable.Builder[A, ConcurrentQueue[A]] =
52     companion.newBuilder[A]
53 
54   /**
55    * Enqueues one element in this queue
56    *
57    * @throws NullPointerException if the specified element is null
58    */
59   def offer(elem: A): Unit = {
60     underlying.offer(elem)
61   }
62 
63   /**
64    * Appends all of the elements in the specified collection to the end of
65    * this queue, in the order that they are returned by the specified
66    * collection's iterator. Attempts to `addAll` of a queue to
67    * itself result in `IllegalArgumentException`.
68    *
69    * @param elems the elements to be inserted into this queue
70    * @return `true` if this queue changed as a result of the call
71    *
72    * @throws `NullPointerException` if the specified collection or any
73    *         of its elements are null
74    */
75   def addAll(elems: A*): Unit = {
76     underlying.addAll(elems.asJava)
77   }
78 
79   /**
80    * Returns the first element in the queue, and removes this element
81    * from the queue.
82    *
83    * @return the first element of the queue or `null` in case the
84    *         queue is empty.
85    */
86   def poll(): A = {
87     underlying.poll()
88   }
89 
90   /**
91    * Removes all of the elements from this queue.
92    * The queue will be empty after this call returns.
93    *
94    * This implementation repeatedly invokes [[poll]] until it
95    * returns `null`.
96    */
97   def clear(): Unit = {
98     underlying.clear()
99   }
100 
101   def iterator: Iterator[A] = {
102     underlying.iterator().asScala
103   }
104 
105   override def seq: Iterable[A] = this
106 
107   override def clone(): ConcurrentQueue[A] = {
108     new ConcurrentQueue[A](toSeq : _*)
109   }
110 
111   override def isEmpty: Boolean = {
112     underlying.isEmpty
113   }
114 
115   override def nonEmpty: Boolean = {
116     !underlying.isEmpty
117   }
118 }
