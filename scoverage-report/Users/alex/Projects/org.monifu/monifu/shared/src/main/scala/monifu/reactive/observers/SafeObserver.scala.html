28 /**
29  * A safe observer ensures too things:
30  *
31  * - errors triggered by downstream observers are caught and streamed to `onError`,
32  *   while the upstream gets an `Ack.Cancel`, to stop sending events
33  *
34  * - once an `onError` or `onComplete` was emitted, the observer no longer accepts
35  *   `onNext` events, ensuring that the Rx grammar is respected.
36  *
37  * - if downstream signals a `Cancel`, the observer no longer accepts any events,
38  *   ensuring that the Rx grammar is respected.
39  *
40  * This implementation doesn't address multi-threading concerns in any way.
41  */
42 final class SafeObserver[-T] private (observer: Observer[T])(implicit s: Scheduler)
43   extends Observer[T] {
44 
45   @volatile
46   private[this] var isDone = false
47 
48   def onNext(elem: T): Future[Ack] = {
49     if (!isDone)
50       try {
51         val r = observer.onNext(elem)
52         onCancelMarkDone(r)
53       }
54       catch {
55         case NonFatal(ex) =>
56           onError(ex)
57           Cancel
58       }
59     else
60       Cancel
61   }
62 
63   def onError(ex: Throwable) = {
64     if (!isDone) {
65       isDone = true
66 
67       try observer.onError(ex) catch {
68         case NonFatal(err) =>
69           s.reportFailure(err)
70       }
71     }
72   }
73 
74   def onComplete() = {
75     if (!isDone) {
76       isDone = true
77 
78       try observer.onComplete() catch {
79         case NonFatal(ex) =>
80           try observer.onError(ex) catch {
81             case NonFatal(err) =>
82               s.reportFailure(ex)
83               s.reportFailure(err)
84           }
85       }
86     }
87   }
88 
89   private[this] def onCancelMarkDone(source: Future[Ack]): Future[Ack] = {
90     source match {
91       case sync if sync.isCompleted =>
92         sync.value.get match {
93           case Continue.IsSuccess =>
94             Continue
95 
96           case Cancel.IsSuccess =>
97             isDone = true
98             Cancel
99 
100           case Failure(ex) =>
101             onError(sync.value.get.failed.get)
102             Cancel
103 
104           case _ =>
105             Continue
106         }
107 
108       case async =>
109         val p = Promise[Ack]()
110 
111         source.onComplete {
112           case Continue.IsSuccess =>
113             p.success(Continue)
114 
115           case Cancel.IsSuccess =>
116             isDone = true
117             p.success(Cancel)
118 
119           case Failure(ex) =>
120             onError(ex)
121             p.success(Cancel)
122 
123           case _ =>
124             p.success(Continue)
125         }
126 
127         p.future
128     }
129   }
130 }
