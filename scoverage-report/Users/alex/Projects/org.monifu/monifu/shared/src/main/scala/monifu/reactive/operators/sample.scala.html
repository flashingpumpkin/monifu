30   /**
31    * Implementation for `Observable.sample(initialDelay, delay)`.
32    *
33    * By comparison with [[monifu.reactive.Observable.sampleRepeated]],
34    * this version does not emit any events if no fresh values were emitted
35    * since the last sampling.
36    */
37   def once[T](source: Observable[T], initialDelay: FiniteDuration, delay: FiniteDuration): Observable[T] =
38     Observable.create { subscriber =>
39       implicit val s = subscriber.scheduler
40       val observer = subscriber.observer
41 
42       source.unsafeSubscribe(new SampleObserver(
43         observer,
44         initialDelay,
45         delay,
46         shouldRepeatOnSilence = false
47       ))
48     }
49 
50   /**
51    * Implementation for `Observable.sampleRepeated(initialDelay, delay)`.
52    *
53    * By comparison with [[monifu.reactive.Observable.sample]], this version always
54    * emits values at the requested interval, even if no fresh value in the meantime.
55    */
56   def repeated[T](source: Observable[T], initialDelay: FiniteDuration, delay: FiniteDuration): Observable[T] =
57      Observable.create { subscriber =>

63         initialDelay,

69   protected[reactive] class SampleObserver[T]
70       (downstream: Observer[T], initialDelay: FiniteDuration, period: FiniteDuration, shouldRepeatOnSilence: Boolean)
71       (implicit s: Scheduler)
72     extends SynchronousObserver[T] {
73 
74     @volatile private[this] var hasValue = false
75     // MUST BE written before `hasValue = true`
76     private[this] var lastValue: T = _
77 
78     // to be written in onComplete/onError, to be read from tick
79     @volatile private[this] var upstreamIsDone = false
80     // MUST BE written to before `upstreamIsDone = true`
81     private[this] var upstreamError: Throwable = null
82 
83     def onNext(elem: T): Ack = {
84       lastValue = elem
85       hasValue = true
86       Continue
87     }
88 
89     def onError(ex: Throwable): Unit = {
90       upstreamError = ex
91       upstreamIsDone = true
92     }
93 
94     def onComplete(): Unit = {
95       upstreamIsDone = true
96     }
97     
98     s.scheduleOnce(initialDelay, new Runnable { self =>
99       import ObserverState.{ON_CONTINUE, ON_NEXT}
100 
101       private[this] val periodMillis = period.toMillis
102       private[this] var nextState = ON_NEXT
103       private[this] var startedAt = 0L
104 
105       def run() = nextState match {
106         case ON_NEXT =>          
107           startedAt = s.currentTimeMillis()
108 
109           if (hasValue) {
110             val result = downstream.onNext(lastValue)
111             hasValue = shouldRepeatOnSilence
112             nextState = ON_CONTINUE
113             result.onContinue(self)
114           }
115           else {
116             scheduleNext(startedAt)
117           }
118 
119         case ON_CONTINUE =>
120           scheduleNext(startedAt)
121       }
122 
123       def scheduleNext(startedAt: Long): Unit = {
124         if (upstreamIsDone) {
125           if (upstreamError != null)
126             downstream.onError(upstreamError)
127           else
128             downstream.onComplete()
129         }
130         else {
131           val nextDelay = {
132             val duration = s.currentTimeMillis() - startedAt
133             val d = periodMillis - duration
134             if (d >= 0L) d else 0L
135           }
136 
137           nextState = ON_NEXT
138           s.scheduleOnce(nextDelay, TimeUnit.MILLISECONDS, self)
139         }
140       }
141     })
142   }
