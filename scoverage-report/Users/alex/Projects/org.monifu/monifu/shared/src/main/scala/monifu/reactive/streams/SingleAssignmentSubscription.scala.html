23 /**
24  * Represents a `org.reactivestreams.Subscription` that can be assigned
25  * only once to another subscription reference.
26  *
27  * If the assignment happens after this subscription has been canceled, then on
28  * assignment the reference will get canceled too. If the assignment
29  * after `request(n)` has been called on this subscription, then
30  * `request(n)` will get called immediately on the assigned reference as well.
31  *
32  * Useful in case you need a thread-safe forward reference.
33  */
34 final class SingleAssignmentSubscription private () extends Subscription {
35   import SingleAssignmentSubscription.State
36   import SingleAssignmentSubscription.State._
37 
38   private[this] val state = Atomic(Empty : State)
39 
40   def :=(s: Subscription): Unit = set(s)
41 
42   def set(s: Subscription): Unit = {
43     val current = state.get
44 
45     current match {
46       case Empty =>
47         if (!state.compareAndSet(current, WithSubscription(s)))
48           set(s) // retry
49 
50       case EmptyRequest(n) =>
51         if (!state.compareAndSet(current, WithSubscription(s)))
52           set(s) // retry
53         else
54           s.request(n)
55 
56       case WithSubscription(_) =>
57         throw new IllegalArgumentException(
58           "SingleAssignmentSubscription as already initialized")
59 
60       case EmptyCanceled =>
61         if (!state.compareAndSet(current, Canceled))
62           set(s) // retry
63         else
64           s.cancel()
65 
66       case Canceled =>
67         throw new IllegalArgumentException(
68           "SingleAssignmentSubscription as already initialized")
69     }
70   }
71 
72   @tailrec
73   def cancel(): Unit = {
74     val current = state.get
75 
76     current match {
77       case Empty =>
78         if (!state.compareAndSet(current, EmptyCanceled))
79           cancel() // retry
80 
81       case EmptyRequest(_) =>
82         if (!state.compareAndSet(current, EmptyCanceled))
83           cancel() // retry
84 
85       case WithSubscription(s) =>
86         if (!state.compareAndSet(current, Canceled))
87           cancel() // retry
88         else
89           s.cancel()
90 
91       case EmptyCanceled | Canceled =>
92         () // do nothing
93     }
94   }
95 
96   @tailrec
97   def request(n: Long): Unit = {
98     val current = state.get
99 
100     current match {
101       case Empty =>
102         if (!state.compareAndSet(current, EmptyRequest(n)))
103           request(n) // retry
104 
105       case EmptyRequest(previous) =>
106         if (!state.compareAndSet(current, EmptyRequest(previous + n)))
107           request(n) // retry
108 
109       case WithSubscription(s) =>
110         s.request(n)
111 
112       case EmptyCanceled | Canceled =>
113         ()
114     }
115   }
116 }
117 
118 object SingleAssignmentSubscription {
119   /** Builder for [[SingleAssignmentSubscription]] */
120   def apply(): SingleAssignmentSubscription =
121     new SingleAssignmentSubscription

123   private sealed trait State
124 
125   private object State {
126     case object Empty extends State
127     case class EmptyRequest(nr: Long) extends State
128     case object EmptyCanceled extends State
129     case class WithSubscription(s: Subscription) extends State
130     case object Canceled extends State
131   }
132 }
