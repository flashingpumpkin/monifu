30    */
31   def onNext[T](source: Observable[T])(cb: T => Unit): Observable[T] =
32     Observable.create[T] { subscriber =>
33       implicit val s = subscriber.scheduler
34       val observer = subscriber.observer
35 
36       source.unsafeSubscribe(new Observer[T] {
37         def onError(ex: Throwable) = observer.onError(ex)
38         def onComplete() = observer.onComplete()
39 
40         def onNext(elem: T) = {
41           // See Section 6.4. in the Rx Design Guidelines:
42           // Protect calls to user code from within an operator
43           var streamError = true
44           try {
45             cb(elem)
46             streamError = false
47             observer.onNext(elem)
48           }
49           catch {
50             case NonFatal(ex) =>
51               if (streamError) { observer.onError(ex); Cancel } else Future.failed(ex)
52           }
53         }
54       })
55     }

129     Observable.create[T] { subscriber =>
130       implicit val s = subscriber.scheduler
131       val observer = subscriber.observer

151    * Implementation for [[Observable.doOnStart]].
152    */
153   def onStart[T](source: Observable[T])(cb: T => Unit): Observable[T] =
154     Observable.create { subscriber =>
155       implicit val s = subscriber.scheduler
156       val observer = subscriber.observer
157 
158       source.unsafeSubscribe(new Observer[T] {
159         private[this] var isStarted = false
160 
161         def onNext(elem: T) = {
162           if (!isStarted) {
163             isStarted = true
164             var streamError = true
165             try {
166               cb(elem)
167               streamError = false
168               observer.onNext(elem)
169             }
170             catch {
171               case NonFatal(ex) =>
172                 observer.onError(ex)
173                 Cancel
174             }
175           }
176           else
177             observer.onNext(elem)
178         }
179 
180         def onError(ex: Throwable) = observer.onError(ex)
181         def onComplete() = observer.onComplete()
182       })
183     }
