29 /**
30  * `ReplaySubject` emits to any observer all of the items that were emitted
31  * by the source, regardless of when the observer subscribes.
32  *
33  * 
34  */

40   private[this] val state = Atomic(Empty(Queue.empty) : State[T])

42   def subscribeFn(subscriber: Subscriber[T]): Unit = {
43     @tailrec
44     def loop(): ConnectableSubscriber[T] = {
45       state.get match {
46         case current @ Empty(cache) =>
47           val connectable = ConnectableSubscriber(subscriber)
48           connectable.pushNext(cache : _*)
49 
50           if (!state.compareAndSet(current, Active(Array(connectable), cache)))
51             loop()
52           else
53             connectable
54 
55         case current @ Active(observers, cache) =>
56           val connectable = ConnectableSubscriber(subscriber)
57           connectable.pushNext(cache : _*)
58 
59           if (!state.compareAndSet(current, Active(observers :+ connectable, cache)))
60             loop()
61           else
62             connectable
63 
64         case current @ Complete(cache, errorThrown) =>
65           val connectable = ConnectableSubscriber(subscriber)
66           connectable.pushNext(cache : _*)
67 
68           if (errorThrown ne null)
69             connectable.pushError(errorThrown)
70           else
71             connectable.pushComplete()
72 
73           connectable
74       }
75     }
76 
77     loop().connect()
78   }

80   private[this] def emitNext(subscriber: Subscriber[T], elem: T): Future[Continue] = {
81     implicit val s = subscriber.scheduler
82     val obs = subscriber.observer
83 
84     obs.onNext(elem) match {
85       case sync if sync.isCompleted =>
86         sync.value.get match {
87           case Continue.IsSuccess =>
88             Continue
89 
90           case _ =>
91             removeSubscription(subscriber)
92             Continue
93         }
94       case async =>
95         async.map {
96           case Continue => Continue
97           case Cancel =>
98             removeSubscription(subscriber)
99             Continue
100         }
101     }
102   }
103 
104   @tailrec
105   def onNext(elem: T): Future[Ack] = {
106     state.get match {
107       case current @ Empty(_) =>
108         if (!state.compareAndSet(current, Empty(current.cache.enqueue(elem))))
109           onNext(elem)
110         else
111           Continue
112 
113       case current @ Active(subscribers, cache) =>
114         if (!state.compareAndSet(current, Active(subscribers, cache.enqueue(elem))))
115           onNext(elem)
116 
117         else {
118           var idx = 0
119           var acc = Continue : Future[Continue]
120 
121           while (idx < subscribers.length) {
122             val subscriber = subscribers(idx)
123             implicit val s = subscriber.scheduler
124 
125             acc = if (acc == Continue || (acc.isCompleted && acc.value.get.isSuccess))
126               emitNext(subscriber, elem)
127             else {
128               val f = emitNext(subscriber, elem)
129               acc.flatMap(_ => f)
130             }
131 
132             idx += 1
133           }
134 
135           acc
136         }
137 
138       case _ =>
139         Cancel
140     }
141   }
142 
143   @tailrec
144   def onComplete(): Unit =
145     state.get match {
146       case current @ Empty(cache) =>
147         if (!state.compareAndSet(current, Complete(cache, null))) {
148           onComplete() // retry
149         }
150       case current @ Active(observers, cache) =>
151         if (!state.compareAndSet(current, Complete(cache, null))) {
152           onComplete() // retry
153         }
154         else {
155           var idx = 0
156           while (idx < observers.length) {
157             observers(idx).observer.onComplete()
158             idx += 1
159           }
160         }
161       case _ =>
162         () // already complete, ignore
163     }
164 
165   @tailrec
166   def onError(ex: Throwable): Unit =
167     state.get match {
168       case current @ Empty(cache) =>
169         if (!state.compareAndSet(current, Complete(cache, ex))) {
170           onError(ex) // retry
171         }
172       case current @ Active(observers, cache) =>
173         if (!state.compareAndSet(current, Complete(cache, ex))) {
174           onError(ex) // retry
175         }
176         else {
177           var idx = 0
178           while (idx < observers.length) {
179             observers(idx).observer.onError(ex)
180             idx += 1
181           }
182         }
183       case _ =>
184         () // already complete, ignore
185     }
186 
187   @tailrec
188   private[this] def removeSubscription(subscriber: Subscriber[T]): Unit = {
189     state.get match {
190       case current @ Active(observers, _) =>
191         val update = current.copy(observers.filterNot(_ eq subscriber))
192         if (!state.compareAndSet(current, update))
193           removeSubscription(subscriber)
194 
195       case _ =>
196         () // not active, do nothing
197     }
198   }
199 }
