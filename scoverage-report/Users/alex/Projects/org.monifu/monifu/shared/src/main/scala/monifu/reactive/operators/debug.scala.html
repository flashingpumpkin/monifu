31    * Implementation for [[Observable.dump]].
32    */
33   def dump[T](source: Observable[T], prefix: String, out: PrintStream): Observable[T] =
34     Observable.create[T] { subscriber =>
35       implicit val s = subscriber.scheduler
36       val observer = subscriber.observer
37 
38       source.unsafeSubscribe(new Observer[T] {
39         private[this] var pos = 0
40         private[this] val downstreamActive = Cancelable {
41           pos += 1; out.println(s"$pos: $prefix canceled")
42         }
43 
44         def onNext(elem: T): Future[Ack] = {
45           // Protects calls to user code from within the operator
46           var streamError = true
47           try {
48             out.println(s"$pos: $prefix-->$elem")
49             streamError = false
50 
51             pos += 1
52             observer.onNext(elem)
53               .ifCanceledDoCancel(downstreamActive)
54           }
55           catch {
56             case NonFatal(ex) =>
57               if (streamError) { observer.onError(ex); Cancel } else
58                 Future.failed(ex)
59           }
60         }
61 
62         def onError(ex: Throwable) = {
63           try {
64             out.println(s"$pos: $prefix-->$ex")
65             pos += 1
66           }
67           catch {
68             case NonFatal(_) =>
69               () // ignore
70           }
71 
72           observer.onError(ex)
73         }
74 
75         def onComplete() = {
76           try {
77             out.println(s"$pos: $prefix completed")
78             pos += 1
79           }
80           catch {
81             case NonFatal(_) =>
82               () // ignore
83           }
84 
85           observer.onComplete()
86         }
87       })
88     }
