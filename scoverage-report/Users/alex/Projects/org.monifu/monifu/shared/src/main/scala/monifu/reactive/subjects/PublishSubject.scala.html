25 /**
26  * A `PublishSubject` emits to a subscriber only those items that are
27  * emitted by the source subsequent to the time of the subscription
28  *
29  * 
30  *
31  * If the source terminates with an error, the `PublishSubject` will not emit any
32  * items to subsequent subscribers, but will simply pass along the error
33  * notification from the source Observable.
34  *
35  * 
36  */
37 final class PublishSubject[T] extends Subject[T,T] {
38   private[this] val lock = new AnyRef
39   private[this] var isCompleted = false
40   private[this] var errorThrown: Throwable = null
41   @volatile private[this] var subscriptions = Array.empty[Subscriber[T]]
42 
43   def subscribeFn(subscriber: Subscriber[T]): Unit =
44     lock.synchronized {
45       if (!isCompleted)
46         subscriptions = createSubscription(subscriptions, subscriber)
47       else if (errorThrown ne null)
48         subscriber.observer.onError(errorThrown)
49       else
50         subscriber.observer.onComplete()
51     }
52 
53   def onNext(elem: T): Future[Ack] = {
54     if (!isCompleted) {
55       val observers = subscriptions
56       if (observers.nonEmpty)
57         streamToMany(observers, elem)
58       else
59         Continue
60     }
61     else
62       Cancel
63   }
64 
65   def onError(ex: Throwable) =
66     lock.synchronized {
67       if (!isCompleted) {
68         isCompleted = true
69         errorThrown = ex
70 
71         var idx = 0
72         while (idx < subscriptions.length) {
73           subscriptions(idx).observer.onError(ex)
74           idx += 1
75         }
76       }
77     }
78 
79   def onComplete() =
80     lock.synchronized {
81       if (!isCompleted) {
82         isCompleted = true
83 
84         var idx = 0
85         while (idx < subscriptions.length) {
86           subscriptions(idx).observer.onComplete()
87           idx += 1
88         }
89       }
90     }
91 
92   private[this] def streamToMany(array: Array[Subscriber[T]], elem: T): Future[Continue] = {
93     val newPromise = PromiseCounter[Continue](Continue, array.length)
94     val length = array.length
95     var idx = 0
96 
97     while (idx < length) {
98       val subscriber = array(idx)
99       implicit val s = subscriber.scheduler
100       val obs = subscriber.observer
101 
102       obs.onNext(elem).onCompleteNow {
103         case Continue.IsSuccess =>
104           newPromise.countdown()
105         case _ =>
106           removeSubscription(subscriber)
107           newPromise.countdown()
108       }
109 
110       idx += 1
111     }
112 
113     newPromise.future
114   }
115 
116   private[this] def removeSubscription(subscriber: Subscriber[T]): Unit =
117     lock.synchronized {
118       subscriptions = subscriptions.filter(_ != subscriber)
119     }
120 
121   private[this] def createSubscription(observers: Array[Subscriber[T]], instance: Subscriber[T]): Array[Subscriber[T]] =
122     lock.synchronized {
123       if (!observers.contains(instance))
124         observers :+ instance
125       else
126         observers
127     }
128 }
