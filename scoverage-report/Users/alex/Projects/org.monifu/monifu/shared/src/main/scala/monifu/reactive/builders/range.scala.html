23 object range {
24   /**
25    * Creates an Observable that emits items in the given range.
26    *
27    * 
28    *
29    * @param from the range start
30    * @param until the range end
31    * @param step increment step, either positive or negative
32    */
33   def apply(from: Long, until: Long, step: Long = 1): Observable[Long] = {
34     require(step != 0, "step must be a number different from zero")
35 
36     Observable.create { subscriber =>
37       implicit val s = subscriber.scheduler
38       val o = subscriber.observer
39 
40       def scheduleLoop(from: Long, until: Long, step: Long): Unit =
41         s.execute(new Runnable {
42           private[this] def isInRange(x: Long): Boolean = {
43             (step > 0 && x < until) || (step < 0 && x > until)
44           }
45 
46           def reschedule(from: Long, until: Long, step: Long): Unit =
47             fastLoop(from, until, step)
48 
49           @tailrec
50           def fastLoop(from: Long, until: Long, step: Long): Unit =
51             if (isInRange(from))
52               o.onNext(from) match {
53                 case sync if sync.isCompleted =>
54                   sync.value.get match {
55                     case Continue.IsSuccess =>
56                       fastLoop(from + step, until, step)
57                     case _ =>
58                       () // do nothing else
59                   }
60 
61                 case async =>
62                   async.onComplete {
63                     case Continue.IsSuccess =>
64                       reschedule(from + step, until, step)
65                     case _ =>
66                       () // do nothing else
67                   }
68               }
69             else
70               o.onComplete()
71 
72           def run(): Unit = {
73             fastLoop(from, until, step)
74           }
75         })
76 
77       scheduleLoop(from, until, step)
78     }
79   }
