29 object timeout {
30   def emitError[T](source: Observable[T], timeout: FiniteDuration): Observable[T] =
31     switchToBackup(source, timeout, Observable.error(
32       new TimeoutException(s"Observable timed-out after $timeout of inactivity")
33     ))
34 
35   def switchToBackup[T](source: Observable[T], timeout: FiniteDuration, backup: Observable[T]) =
36     Observable.create[T] { subscriber =>
37       implicit val s = subscriber.scheduler
38 
39       source.unsafeSubscribe(new Observer[T] with Runnable { self =>
40         private[this] val downstream = subscriber.observer
41         private[this] val timeoutMillis = timeout.toMillis
42         private[this] val task = MultiAssignmentCancelable()
43         private[this] var ack: Future[Ack] = Continue
44         // MUST BE synchronized by `self`
45         private[this] var isDone = false
46         // MUST BE synchronized by `self`
47         private[this] var lastEmittedMillis: Long = s.currentTimeMillis()
48 
49         locally {
50           task := s.scheduleOnce(timeout, self)
51         }
52 
53         def run(): Unit = self.synchronized {
54           if (!isDone) {
55             val rightNow = s.currentTimeMillis()
56             val sinceLastOnNextInMillis = rightNow - lastEmittedMillis
57 
58             if (sinceLastOnNextInMillis >= timeoutMillis) {
59               isDone = true
60               ack.onContinue {
61                 // subscribing our downstream observer to the backup observable
62                 backup.unsafeSubscribe(subscriber)
63               }
64             }
65             else {
66               val remainingTimeMillis = timeoutMillis - sinceLastOnNextInMillis
67               task := s.scheduleOnce(remainingTimeMillis, TimeUnit.MILLISECONDS, self)
68             }
69           }
70         }
71 
72         def onNext(elem: T): Future[Ack] = {
73           // unfortunately we can't get away of this synchronization
74           // as the scheduler is contending on the same downstream
75           self.synchronized {
76             if (isDone) Cancel else {
77               lastEmittedMillis = s.currentTimeMillis()
78               ack = downstream.onNext(elem)
79               ack
80             }
81           }
82         }
83 
84         def onError(ex: Throwable) = self.synchronized {
85           if (!isDone) {
86             isDone = true
87             ack.onContinueSignalError(downstream, ex)
88           }
89         }
90 
91         def onComplete() = self.synchronized {
92           if (!isDone) {
93             isDone = true
94             ack.onContinueSignalComplete(downstream)
95           }
96         }
97       })
98     }
99 }
