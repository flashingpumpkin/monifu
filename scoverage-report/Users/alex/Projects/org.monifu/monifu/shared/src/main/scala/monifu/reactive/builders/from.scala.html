30    * Implementation for [[Observable.fromIterable]].
31    */
32   def iterable[T](iterable: Iterable[T]): Observable[T] = {
33     Observable.create { subscriber =>
34       implicit val s = subscriber.scheduler
35       val observer = subscriber.observer
36 
37       def startFeedLoop(iterator: Iterator[T]): Unit = s.execute(new Runnable {
38         /**
39          * Loops synchronously, pushing elements into the downstream observer,
40          * until either the iterator is finished or an asynchronous barrier
41          * is being hit.
42          */
43         @tailrec
44         def fastLoop(): Unit = {
45           // the result of onNext calls, on which we must do back-pressure
46           var ack: Future[Ack] = Continue
47           // we do not want to catch errors from our interaction with our observer,
48           // since SafeObserver should take care of than, hence we must only
49           // catch and stream errors related to the interactions with the iterator
50           var streamError = true
51           // true in case our iterator is seen to be empty and we must signal onComplete
52           var signalDone = false
53           // non-null in case we caught an iterator related error and we must signal onError
54           var signalError: Throwable = null
55 
56           try {
57             if (iterator.hasNext) {
58               val next = iterator.next()
59               streamError = false
60               ack = observer.onNext(next)
61             }
62             else
63               signalDone = true
64           }
65           catch {
66             case NonFatal(ex) if streamError =>
67               signalError = ex
68           }
69 
70           if (signalDone) {
71             observer.onComplete()
72           }
73           else if (signalError != null) {
74             observer.onError(signalError)
75           }
76           else
77             ack match {
78               case sync if sync.isCompleted =>
79                 sync.value.get match {
80                   case Continue.IsSuccess =>
81                     fastLoop()
82                   case _ =>
83                     () // do nothing
84                 }
85 
86               case async =>
87                 async.onComplete {
88                   case Continue.IsSuccess =>
89                     run()
90                   case _ =>
91                     () // do nothing
92                 }
93             }
94         }
95 
96         def run(): Unit = {
97           fastLoop()
98         }
99       })
100 
101       var streamError = true
102       try {
103         val iterator = iterable.iterator
104         val isEmpty = iterator.isEmpty
105         streamError = false
106 
107         if (isEmpty)
108           observer.onComplete()
109         else
110           startFeedLoop(iterator)
111       }
112       catch {
113         case NonFatal(ex) if streamError =>
114           observer.onError(ex)
115       }
116     }
117   }
118 
119   /**
120    * Implementation for [[Observable.fromFuture]].
121    */
122   def future[T](f: Future[T]): Observable[T] =
123     Observable.create { subscriber =>
124       implicit val s = subscriber.scheduler
125       val o = subscriber.observer
126 
127       f.value match {
128         case Some(Success(value)) =>
129           o.onNext(value).onContinueSignalComplete(o)
130         case Some(Failure(ex)) =>
131           o.onError(ex)
132         case None => f.onComplete {
133           case Success(value) =>
134             o.onNext(value).onContinueSignalComplete(o)
135           case Failure(ex) =>
136             o.onError(ex)
137         }
138       }
139     }
