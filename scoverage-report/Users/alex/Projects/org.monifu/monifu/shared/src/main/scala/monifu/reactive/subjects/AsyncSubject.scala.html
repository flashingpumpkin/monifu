29 /**
30  * An `AsyncSubject` emits the last value (and only the last value) emitted by the source Observable,
31  * and only after that source Observable completes.
32  *
33  * 
34  *
35  * If the source terminates with an error, the `AsyncSubject` will not emit any
36  * items to subsequent subscribers, but will simply pass along the error
37  * notification from the source Observable.
38  *
39  * 
40  */
41 final class AsyncSubject[T] extends Subject[T,T] { self =>
42   import monifu.reactive.subjects.AsyncSubject._
43 
44   private[this] val state = Atomic(Active(Set.empty[Subscriber[T]]) : State[T])
45   private[this] var onNextHappened = false
46   private[this] var currentElem: T = _
47   private[this] var isCompleted = false
48 
49   @tailrec
50   def subscribeFn(subscriber: Subscriber[T]): Unit =
51     state.get match {
52       case current @ Active(set) =>
53         if (!state.compareAndSet(current, Active(set + subscriber)))
54           subscribeFn(subscriber)
55       case CompletedEmpty =>
56         subscriber.observer.onComplete()
57       case CompletedError(ex) =>
58         subscriber.observer.onError(ex)
59       case Completed(value) =>
60         implicit val s = subscriber.scheduler
61         subscriber.observer.onNext(value)
62           .onContinueSignalComplete(subscriber.observer)
63     }
64 
65   def onNext(elem: T): Future[Ack] = {
66     if (!isCompleted) {
67       if (!onNextHappened) onNextHappened = true
68       currentElem = elem
69       Continue
70     }
71     else
72       Cancel
73   }
74 
75   @tailrec
76   def onError(ex: Throwable): Unit = state.get match {
77     case current @ Active(set) =>
78       isCompleted = true
79       if (!state.compareAndSet(current, CompletedError(ex)))
80         onError(ex)
81       else
82         for (s <- set) s.observer.onError(ex)
83 
84     case _ =>
85       () // already completed, do nothing
86   }
87 
88   @tailrec
89   def onComplete(): Unit = state.get match {
90     case current @ Active(set) =>
91       isCompleted = true
92 
93       if (onNextHappened)
94         if (!state.compareAndSet(current, Completed(currentElem)))
95           onComplete()
96         else
97           for (subscriber <- set) {
98             implicit val s = subscriber.scheduler
99             subscriber.observer.onNext(currentElem)
100               .onContinueSignalComplete(subscriber.observer)
101           }
102       else if (!state.compareAndSet(current, CompletedEmpty))
103         onComplete()
104       else
105         for (s <- set) s.observer.onComplete()
106 
107     case _ =>
108       () // already completed, do nothing
109   }
110 }

119   private case class Completed[+T](value: T) extends State[T]
120   private case class CompletedError(ex: Throwable) extends State[Nothing]
121 }
