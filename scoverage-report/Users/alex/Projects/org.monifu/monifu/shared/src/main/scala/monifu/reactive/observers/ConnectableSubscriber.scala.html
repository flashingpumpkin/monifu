27 /**
28  * Wraps an [[Observer]] into an implementation that abstains from emitting items until the call
29  * to `connect()` happens. Prior to `connect()` it's also a [[Channel]] into which you can enqueue
30  * events for delivery once `connect()` happens, but before any items
31  * emitted by `onNext` / `onComplete` and `onError`.
32  *
33  * Example: {{{
34  *   val obs = ConnectableObserver(observer)
35  *
36  *   // schedule onNext event, after connect()
37  *   obs.onNext("c")
38  *
39  *   // schedule event "a" to be emitted first
40  *   obs.pushNext("a")
41  *   // schedule event "b" to be emitted second
42  *   obs.pushNext("b")
43  *
44  *   // underlying observer now gets events "a", "b", "c" in order
45  *   obs.connect()
46  * }}}
47  *
48  * Example of an observer ended in error: {{{
49  *   val obs = ConnectableObserver(observer)
50  *
51  *   // schedule onNext event, after connect()
52  *   obs.onNext("c")
53  *
54  *   obs.pushNext("a") // event "a" to be emitted first
55  *   obs.pushNext("b") // event "b" to be emitted first
56  *
57  *   // schedule an onError sent downstream, once connect()
58  *   // happens, but after "a" and "b"
59  *   obs.pushError(new RuntimeException())
60  *
61  *   // underlying observer receives ...
62  *   // onNext("a") -> onNext("b") -> onError(RuntimeException)
63  *   obs.connect()
64  *
65  *   // NOTE: that onNext("c") never happens
66  * }}}
67  */

85   // Promise guaranteed to be fulfilled once isConnected is
86   // seen as true and used for back-pressure.
87   // MUST BE synchronized by `lock`, only available if isConnected == false
88   private[this] val connectedPromise = Promise[Ack]()

91   // Volatile that is set to true once the buffer is drained.
92   // Once visible as true, it implies that the queue is empty
93   // and has been drained and thus the onNext/onError/onComplete
94   // can take the fast path
95   @volatile private[this] var isConnected = false

97   /**
98    * Connects the underling observer to the upstream publisher.
99    *
100    * This function should be idempotent. Calling it multiple times should have the same
101    * effect as calling it once.
102    */
103   def connect(): Unit =
104     lock.synchronized {
105       if (!isConnected && !isConnectionStarted) {
106         isConnectionStarted = true
107 
108         Observable.fromIterable(queue).unsafeSubscribe(new Observer[T] {
109           private[this] val bufferWasDrained = Promise[Ack]()
110 
111           bufferWasDrained.future.onSuccess {
112             case Continue =>
113               connectedPromise.success(Continue)
114               isConnected = true
115               queue = null // gc relief
116 
117             case Cancel =>
118               wasCanceled = true
119               connectedPromise.success(Cancel)
120               isConnected = true
121               queue = null // gc relief
122           }
123 
124           def onNext(elem: T): Future[Ack] = {
125             underlying.onNext(elem)
126               .ifCancelTryCanceling(bufferWasDrained)
127           }
128 
129           def onComplete(): Unit = {
130             if (!scheduledDone) {
131               bufferWasDrained.trySuccess(Continue)
132             }
133             else if (scheduledError ne null) {
134               if (bufferWasDrained.trySuccess(Cancel))
135                 underlying.onError(scheduledError)
136             }
137             else if (bufferWasDrained.trySuccess(Cancel))
138               underlying.onComplete()
139           }
140 
141           def onError(ex: Throwable): Unit = {
142             if (scheduledError ne null)
143               s.reportFailure(ex)
144             else {
145               scheduledDone = true
146               scheduledError = ex
147 
148               if (bufferWasDrained.trySuccess(Cancel))
149                 underlying.onError(ex)
150               else
151                 s.reportFailure(ex)
152             }
153           }
154         })
155       }
156     }
157 
158   /**
159    * Emit an item immediately to the underlying observer,
160    * after previous `pushNext()` events, but before any events emitted through
161    * `onNext`.
162    */
163   def pushNext(elems: T*) =
164     lock.synchronized {
165       if (isConnected || isConnectionStarted)
166         throw new IllegalStateException("Observer was already connected, so cannot pushNext")
167       else if (!scheduledDone)
168         queue.append(elems : _*)
169     }
170 
171   /**
172    * Emit an item
173    */
174   def pushComplete() =
175     lock.synchronized {
176       if (isConnected || isConnectionStarted)
177         throw new IllegalStateException("Observer was already connected, so cannot pushNext")
178       else if (!scheduledDone) {
179         scheduledDone = true
180       }
181     }
182 
183   def pushError(ex: Throwable) =
184     lock.synchronized {
185       if (isConnected || isConnectionStarted)
186         throw new IllegalStateException("Observer was already connected, so cannot pushNext")
187       else if (!scheduledDone) {
188         scheduledDone = true
189         scheduledError = ex
190       }
191     }
192 
193   val observer: Observer[T] = new Observer[T] {
194     def onNext(elem: T) = {
195       if (!isConnected) {
196         // no need for synchronization here, since this reference is initialized
197         // before the subscription happens and because it gets written only in
198         // onNext / onComplete, which are non-concurrent clauses
199         connectedFuture = connectedFuture.flatMap {
200           case Cancel => Cancel
201           case Continue =>
202             underlying.onNext(elem)
203         }
204         connectedFuture
205       }
206       else if (!wasCanceled) {
207         // taking fast path
208         underlying.onNext(elem)
209       }
210       else {
211         // was canceled either during connect, or the upstream publisher
212         // sent an onNext event after onComplete / onError
213         Cancel
214       }
215     }
216 
217     def onComplete() = {
218       // we cannot take a fast path here
219       connectedFuture.onContinueSignalComplete(underlying)
220     }
221 
222     def onError(ex: Throwable) = {
223       // we cannot take a fast path here
224       connectedFuture.onContinueSignalError(underlying, ex)
225     }
226   }
227 }
