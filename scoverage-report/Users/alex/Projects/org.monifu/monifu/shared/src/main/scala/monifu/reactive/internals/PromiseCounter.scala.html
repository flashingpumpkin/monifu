19 import scala.concurrent.Promise
20 import monifu.concurrent.atomic.padded.Atomic

22 /**
23  * Represents a Promise that completes with `value` after
24  * receiving a `countdownUntil` number of `countdown()` calls.
25  */
26 final class PromiseCounter[T] private (value: T, countdownUntil: Int) {
27   require(countdownUntil > 0, "countdownUntil must be strictly positive")
28 
29   private[this] val promise = Promise[T]()
30   private[this] val counter = Atomic(0)
31 
32   def future = promise.future
33 
34   def countdown(): Unit =
35     if (counter.incrementAndGet() >= countdownUntil) {
36       promise.success(value)
37     }
38 
39   def success(value: T) = {
40     promise.success(value)
41   }
42 }
43 
44 object PromiseCounter {
45   def apply[T](value: T, countDown: Int): PromiseCounter[T] =
46     new PromiseCounter[T](value, countDown)
47 }
