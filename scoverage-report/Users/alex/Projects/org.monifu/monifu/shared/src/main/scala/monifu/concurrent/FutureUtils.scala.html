24 object FutureUtils {
25   /**
26    * Utility that returns a new Future that either completes with
27    * the original Future's result or with a TimeoutException in case
28    * the maximum wait time was exceeded.
29    *
30    * @param atMost specifies the maximum wait time until the future is
31    *               terminated with a TimeoutException
32    *
33    * @param s is the Scheduler, needed for completing our internal promise
34    *
35    * @return a new future that will either complete with the result of our
36    *         source or fail in case the timeout is reached.
37    */
38   def withTimeout[T](source: Future[T], atMost: FiniteDuration)(implicit s: Scheduler): Future[T] = {
39     val err = new TimeoutException
40     val promise = Promise[T]()
41     val task = s.scheduleOnce(atMost)(promise.tryFailure(err))
42 
43     source.onComplete { case r =>
44       // canceling task to prevent waisted CPU resources and memory leaks
45       // if the task has been executed already, this has no effect
46       task.cancel()
47       promise.tryComplete(r)
48     }
49 
50     promise.future
51   }
52 
53   /**
54    * Utility that lifts a `Future[T]` into a `Future[Try[T]]`, just because
55    * it is useful sometimes.
56    */
57   def liftTry[T](source: Future[T])(implicit ec: ExecutionContext): Future[Try[T]] = {
58     if (source.isCompleted) {
59       Future.successful(source.value.get)
60     }
61     else {
62       val p = Promise[Try[T]]()
63       source.onComplete { case result => p.success(result) }
64       p.future
65     }
66   }
67 
68   /**
69    * Returns a new `Future` that takes a minimum amount of time to execute,
70    * specified by `atLeast`.
71    *
72    * @param atLeast the minimal duration that the returned future will take to complete.
73    * @param s the implicit scheduler that handles the scheduling and the execution
74    * @return a new `Future` whose execution time is within the specified bounds
75    */
76   def withMinDuration[T](source: Future[T], atLeast: FiniteDuration)(implicit s: Scheduler): Future[T] = {
77     val start = s.currentTimeMillis()
78     val p = Promise[T]()
79 
80     source.onComplete {
81       case result =>
82         val remainingMillis = atLeast.toMillis - (s.currentTimeMillis() - start)
83         if (remainingMillis >= 1)
84           s.scheduleOnce(remainingMillis, TimeUnit.MILLISECONDS)(p.complete(result))
85         else
86           p.complete(result)
87     }
88 
89     p.future
90   }
91 
92   /**
93    * Creates a future that completes with the specified `result`, but only
94    * after the specified `delay`.
95    */
96   def delayedResult[T](delay: FiniteDuration)(result: => T)(implicit s: Scheduler): Future[T] = {
97     val p = Promise[T]()
98     s.scheduleOnce(delay)(p.complete(Try(result)))
99     p.future
100   }
