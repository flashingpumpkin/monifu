31 /**
32  * The Observer from the Rx pattern is the trio of callbacks that
33  * get subscribed to an Observable for receiving events.
34  *
35  * The events received must follow the Rx grammar, which is:
36  *      onNext *   (onComplete | onError)?
37  *
38  * That means an Observer can receive zero or multiple events, the stream
39  * ending either in one or zero `onComplete` or `onError` (just one, not both),
40  * and after onComplete or onError, a well behaved Observable implementation
41  * shouldn't send any more onNext events.
42  */
43 trait Observer[-T] {
44   def onNext(elem: T): Future[Ack]
45 
46   def onError(ex: Throwable): Unit
47 
48   def onComplete(): Unit
49 }
50 
51 object Observer {
52   /**
53    * Given an `org.reactivestreams.Subscriber` as defined by the [[http://www.reactive-streams.org/ Reactive Streams]]
54    * specification, it builds an [[Observer]] instance compliant with the Monifu Rx implementation.
55    */
56   def from[T](subscriber: RSubscriber[T])(implicit s: Scheduler): Observer[T] = {
57     SubscriberAsObserver(subscriber)
58   }
59 
60   /**
61    * Transforms the source [[Observer]] into a `org.reactivestreams.Subscriber`
62    * instance as defined by the [[http://www.reactive-streams.org/ Reactive Streams]]
63    * specification.
64    */
65   def asSubscriber[T](observer: Observer[T], requestSize: Int = 128)(implicit s: Scheduler): RSubscriber[T] = {
66     observer match {
67       case sync: SynchronousObserver[_] =>
68         val inst = sync.asInstanceOf[SynchronousObserver[T]]
69         SynchronousSubscriberAsReactiveSubscriber(SynchronousSubscriber(inst, s), requestSize)
70       case async =>
71         SubscriberAsReactiveSubscriber(Subscriber(async, s), requestSize)
72     }
73   }
74 
75   /**
76    * Implicit conversion from [[Observer]] to `org.reactivestreams.Subscriber`.
77    */
78   def ObserverIsSubscriber[T](source: Observer[T])(implicit s: Scheduler): RSubscriber[T] =
79     Observer.asSubscriber(source)
80 
81   /**
82    * Feeds the given [[Observer]] instance with elements from the given iterable,
83    * respecting the contract and returning a `Future[Ack]` with the last
84    * acknowledgement given after the last emitted element.
85    */
86   def feed[T](observer: Observer[T], iterable: Iterable[T])(implicit s: Scheduler): Future[Ack] = {
87     def scheduleFeedLoop(promise: Promise[Ack], iterator: Iterator[T]): Future[Ack] = {
88       s.execute(new Runnable {
89         @tailrec
90         def fastLoop(): Unit = {
91           val ack = observer.onNext(iterator.next())
92 
93           if (iterator.hasNext)
94             ack match {
95               case sync if sync.isCompleted =>
96                 if (sync == Continue || sync.value.get == Continue.IsSuccess)
97                   fastLoop()
98                 else
99                   promise.completeWith(sync)
100               case async =>
101                 async.onComplete {
102                   case Success(Continue) =>
103                     scheduleFeedLoop(promise, iterator)
104                   case Success(Cancel) =>
105                     promise.success(Cancel)
106                   case Failure(ex) =>
107                     promise.failure(ex)
108                 }
109             }
110           else
111             promise.completeWith(ack)
112         }
113 
114         def run(): Unit = {
115           try fastLoop() catch {
116             case NonFatal(ex) =>
117               try observer.onError(ex) finally {
118                 promise.failure(ex)
119               }
120           }
121         }
122       })
123 
124       promise.future
125     }
126 
127     val iterator = iterable.iterator
128     if (iterator.hasNext) scheduleFeedLoop(Promise[Ack](), iterator) else Continue
129   }
130 }
