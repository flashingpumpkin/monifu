24 final class AsyncQueue[T] private (elems: T*) {
25   private[this] val state = Atomic(State(Queue(elems : _*), Queue.empty))
26 
27   def poll(): Future[T] =
28     state.transformAndExtract {
29       case State(elements, promises) =>
30         if (elements.nonEmpty) {
31           val (e, newQ) = elements.dequeue
32           (Future.successful(e), State(newQ, promises))
33         }
34         else {
35           val p = Promise[T]()
36           (p.future, State(elements, promises.enqueue(p)))
37         }
38     }
39 
40   def offer(elem: T): Unit = {
41     val p = state.transformAndExtract {
42       case State(elements, promises) =>
43         if (promises.nonEmpty) {
44           val (p, q) = promises.dequeue
45           (Some(p), State(elements, q))
46         }
47         else
48           (None, State(elements.enqueue(elem), promises))
49     }
50 
51     p.foreach(_.success(elem))
52   }
53   
54   def clear(): Unit =
55     state.set(State(Queue.empty, Queue.empty))
56 
57   def clearAndOffer(elem: T): Unit =
58     state.set(State(Queue(elem), Queue.empty))
59 
60   private[this] case class State(elements: Queue[T], promises: Queue[Promise[T]])
61 }
