29 /**
30  * A highly optimized [[BufferedSubscriber]] implementation. It supports 2
31  * [[monifu.reactive.BufferPolicy buffer policies]] - unbounded or bounded and terminated
32  * with a [[monifu.reactive.BufferOverflowException BufferOverflowException]].
33  *
34  * To create an instance using an unbounded policy: {{{
35  *   // by default, the constructor for BufferedSubscriber is returning this unbounded variant
36  *   BufferedSubscriber(observer)
37  *
38  *   // or you can specify the Unbounded policy explicitly
39  *   import monifu.reactive.BufferPolicy.Unbounded
40  *   val buffered = BufferedSubscriber(observer, bufferPolicy = Unbounded)
41  * }}}
42  *
43  * To create a bounded buffered observable that triggers
44  * [[monifu.reactive.BufferOverflowException BufferOverflowException]]
45  * when over capacity: {{{
46  *   import monifu.reactive.BufferPolicy.OverflowTriggering
47  *   // triggers buffer overflow error after 10000 messages
48  *   val buffered = BufferedSubscriber(observer, bufferPolicy = OverflowTriggering(bufferSize = 10000))
49  * }}}
50  *
51  * @param underlying is the underlying observer receiving the queued events
52  * @param bufferSize is the maximum buffer size, or zero if unbounded
53  */
54 final class SynchronousBufferedSubscriber[-T] private
55     (underlying: Observer[T], bufferSize: Int = 0)(implicit val scheduler: Scheduler)
56   extends BufferedSubscriber[T] with SynchronousSubscriber[T] {
57 
58   require(bufferSize >= 0, "bufferSize must be a positive number")
59 
60   private[this] val queue = ConcurrentQueue.empty[T]
61   // to be modified only in onError, before upstreamIsComplete
62   private[this] var errorThrown: Throwable = null
63   // to be modified only in onError / onComplete
64   @volatile private[this] var upstreamIsComplete = false
65   // to be modified only by consumer
66   @volatile private[this] var downstreamIsDone = false
67   // for enforcing non-concurrent updates
68   private[this] val itemsToPush = Atomic(0)
69 
70   val observer: SynchronousObserver[T] = new SynchronousObserver[T] {
71     def onNext(elem: T): Ack = {
72       if (!upstreamIsComplete && !downstreamIsDone) {
73         try {
74           queue.offer(elem)
75           pushToConsumer()
76           Continue
77         }
78         catch {
79           case NonFatal(ex) =>
80             onError(ex)
81             Cancel
82         }
83       }
84       else
85         Cancel
86     }
87 
88     def onError(ex: Throwable) = {
89       if (!upstreamIsComplete && !downstreamIsDone) {
90         errorThrown = ex
91         upstreamIsComplete = true
92         pushToConsumer()
93       }
94     }
95 
96     def onComplete() = {
97       if (!upstreamIsComplete && !downstreamIsDone) {
98         upstreamIsComplete = true
99         pushToConsumer()
100       }
101     }
102   }
103 
104 
105   @tailrec
106   private[this] def pushToConsumer(): Unit = {
107     val currentNr = itemsToPush.get
108 
109     if (bufferSize == 0) {
110       // unbounded branch
111       if (!itemsToPush.compareAndSet(currentNr, currentNr + 1))
112         pushToConsumer()
113       else if (currentNr == 0)
114         scheduler.execute(new Runnable {
115           def run() = fastLoop(0)
116         })
117     }
118     else {
119       // triggering overflow branch
120       if (currentNr >= bufferSize && !upstreamIsComplete) {
121         observer.onError(new BufferOverflowException(
122           s"Downstream observer is too slow, buffer over capacity with a specified buffer size of $bufferSize and" +
123             s" $currentNr events being left for push"))
124       }
125       else if (!itemsToPush.compareAndSet(currentNr, currentNr + 1))
126         pushToConsumer()
127       else if (currentNr == 0)
128         scheduler.execute(new Runnable {
129           def run() = fastLoop(0)
130         })
131     }
132   }
133 
134   private[this] def rescheduled(processed: Int): Unit = {
135     fastLoop(processed)
136   }
137 
138   @tailrec
139   private[this] def fastLoop(processed: Int): Unit = {
140     if (!downstreamIsDone) {
141       val hasError = errorThrown ne null
142       val next = queue.poll()
143 
144       if (next != null) {
145         underlying.onNext(next) match {
146           case sync if sync.isCompleted =>
147             sync match {
148               case continue if continue == Continue || continue.value.get == Continue.IsSuccess =>
149                 // process next
150                 fastLoop(processed + 1)
151 
152               case done if done == Cancel || done.value.get == Cancel.IsSuccess =>
153                 // ending loop
154                 downstreamIsDone = true
155                 itemsToPush.set(0)
156 
157               case error if error.value.get.isFailure =>
158                 // ending loop
159                 downstreamIsDone = true
160                 itemsToPush.set(0)
161                 underlying.onError(error.value.get.failed.get)
162             }
163 
164           case async =>
165             async.onComplete {
166               case Continue.IsSuccess =>
167                 // re-run loop (in different thread)
168                 rescheduled(processed + 1)
169 
170               case Cancel.IsSuccess =>
171                 // ending loop
172                 downstreamIsDone = true
173                 itemsToPush.set(0)
174 
175               case Failure(ex) =>
176                 // ending loop
177                 downstreamIsDone = true
178                 itemsToPush.set(0)
179                 underlying.onError(ex)
180 
181               case other =>
182                 // never happens, but to appease Scala's compiler
183                 downstreamIsDone = true
184                 itemsToPush.set(0)
185                 underlying.onError(new MatchError(s"$other"))
186             }
187         }
188       }
189       else if (upstreamIsComplete || hasError) {
190         // Race-condition check, but if upstreamIsComplete=true is visible,
191         // then the queue should be fully published because there's a clear happens-before
192         // relationship between queue.offer() and upstreamIsComplete=true
193         if (!queue.isEmpty) {
194           fastLoop(processed)
195         }
196         else {
197           // ending loop
198           downstreamIsDone = true
199           itemsToPush.set(0)
200           queue.clear() // for GC purposes
201           if (errorThrown ne null)
202             underlying.onError(errorThrown)
203           else
204             underlying.onComplete()
205         }
206       }
207       else {
208         val remaining = itemsToPush.decrementAndGet(processed)
209         // if the queue is non-empty (i.e. concurrent modifications just happened)
210         // then start all over again
211         if (remaining > 0) fastLoop(0)
212       }
213     }
214   }
215 }
