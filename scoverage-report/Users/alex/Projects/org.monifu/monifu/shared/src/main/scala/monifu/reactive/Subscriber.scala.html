17 package monifu.reactive
18 
19 import monifu.concurrent.Scheduler
20 
21 /**
22  * A `Subscriber` value is a named tuple of an observer and a scheduler,
23  * whose usage is in [[Observable.create]].
24  *
25  * An `Observable.subscribe` takes as parameters both an [[Observer]]
26  * and a [[Scheduler]] and the purpose of a `Subscriber` is convenient
27  * grouping in `Observable.create`.
28  *
29  * A `Subscriber` value is basically an address that the data source needs
30  * in order to send events.
31  */
32 trait Subscriber[-T] {
33   def observer: Observer[T]
34   def scheduler: Scheduler
35 }
36 
37 object Subscriber {
38   /** Subscriber builder */
39   def apply[T](observer: Observer[T], scheduler: Scheduler): Subscriber[T] =
40     new Implementation[T](observer, scheduler)
41 
42   /** Utility for pattern matching */
43   def unapply[T](ref: Subscriber[T]): Some[(Observer[T], Scheduler)] =
44     Some((ref.observer, ref.scheduler))
45 
46   private[this] final class Implementation[-T]
47       (val observer: Observer[T], val scheduler: Scheduler)
48     extends Subscriber[T] {
49 
50     require(observer != null, "Observer should not be null")
51     require(scheduler != null, "Scheduler should not be null")
52 
53     override def equals(other: Any): Boolean = other match {
54       case that: Implementation[_] =>
55         observer == that.observer &&
56         scheduler == that.scheduler
57       case _ =>
58         false
59     }
60 
61     override val hashCode: Int = {
62       31 * observer.hashCode() + scheduler.hashCode()
63     }
64   }
65 }
