31   def concat[T,U](source: Observable[T], delayErrors: Boolean)(implicit ev: T <:< Observable[U]): Observable[U] = {
32     Observable.create[U] { subscriber =>
33       implicit val s = subscriber.scheduler
34       val observerU = subscriber.observer
35 
36       source.unsafeSubscribe(new Observer[T] {
37         private[this] val errors = if (delayErrors)
38           mutable.ArrayBuffer.empty[Throwable] else null
39 
40         private[this] val refCount = RefCountCancelable {
41           if (delayErrors && errors.nonEmpty)
42             observerU.onError(CompositeException(errors))
43           else
44             observerU.onComplete()
45         }
46 
47         def onNext(childObservable: T) = {
48           val upstreamPromise = Promise[Ack]()
49           val refID = refCount.acquire()
50 
51           childObservable.unsafeSubscribe(new Observer[U] {
52             def onNext(elem: U) = {
53               observerU.onNext(elem)
54                 .ifCancelTryCanceling(upstreamPromise)
55             }
56 
57             def onError(ex: Throwable): Unit = {
58               if (delayErrors) {
59                 errors += ex
60                 onComplete()
61               }
62               else {
63                 // error happened, so signaling both the main thread that it should stop
64                 // and the downstream consumer of the error
65                 upstreamPromise.trySuccess(Cancel)
66                 observerU.onError(ex)
67               }
68             }
69 
70             def onComplete(): Unit = {
71               // NOTE: we aren't sending this onComplete signal downstream to our observerU
72               // instead we are just instructing upstream to send the next observable
73               upstreamPromise.trySuccess(Continue)
74               refID.cancel()
75             }
76           })
77 
78           upstreamPromise.future
79         }
80 
81         def onError(ex: Throwable) = {
82           if (delayErrors) {
83             errors += ex
84             onComplete()
85           }
86           else {
87             // oops, error happened on main thread, piping that
88             // along should cancel everything
89             observerU.onError(ex)
90           }
91         }
92 
93         def onComplete() = {
94           refCount.cancel()
95         }
96       })
97     }
98   }

100   /**
101    * Implementation for [[Observable.merge]].
102    */
103   def merge[T,U](source: Observable[T], bufferPolicy: BufferPolicy[U], delayErrors: Boolean)
104       (implicit ev: T <:< Observable[U]): Observable[U] = {
105 
106     Observable.create { subscriber =>
107       implicit val s = subscriber.scheduler
108       val buffer = BufferedSubscriber(subscriber.observer, bufferPolicy)
109       val observerU = buffer.observer
110 
111       source.unsafeSubscribe(new SynchronousObserver[T] {
112         private[this] val streamActivity =
113           BooleanCancelable()
114         private[this] val errors = if (delayErrors)
115           mutable.ArrayBuffer.empty[Throwable] else null
116 
117         private[this] val refCount = RefCountCancelable {
118           if (delayErrors)
119             errors.synchronized {
120               if (errors.nonEmpty)
121                 observerU.onError(CompositeException(errors))
122               else
123                 observerU.onComplete()
124             }
125           else
126             observerU.onComplete()
127         }
128 
129         def onNext(childObservable: T) = {
130           val refID = refCount.acquire()
131 
132           childObservable.unsafeSubscribe(new Observer[U] {
133             def onNext(elem: U) = {
134               if (streamActivity.isCanceled)
135                 Cancel
136               else
137                 observerU.onNext(elem)
138                   .ifCanceledDoCancel(streamActivity)
139             }
140 
141             def onError(ex: Throwable): Unit = {
142               if (delayErrors) errors.synchronized {
143                 errors += ex
144                 refID.cancel()
145               }
146               else {
147                 streamActivity.cancel()
148                 observerU.onError(ex)
149               }
150             }
151 
152             def onComplete(): Unit = {
153               // NOTE: we aren't sending this onComplete signal downstream to our observerU
154               // we will eventually do that after all of them are complete
155               refID.cancel()
156             }
157           })
158 
159           if (streamActivity.isCanceled)
160             Cancel
161           else
162             Continue
163         }
164 
165         def onError(ex: Throwable) = {
166           if (delayErrors) errors.synchronized {
167             errors += ex
168             refCount.cancel()
169           }
170           else {
171             // oops, error happened on main thread,
172             // piping that along should cancel everything
173             observerU.onError(ex)
174           }
175         }
176 
177         def onComplete() = {
178           refCount.cancel()
179         }
180       })
181     }
182   }
