29 /**
30  * `BehaviorSubject` when subscribed, will emit the most recently emitted item by the source,
31  * or the `initialValue` (as the seed) in case no value has yet been emitted, then continuing
32  * to emit events subsequent to the time of invocation.
33  *
34  * 
35  *
36  * When the source terminates in error, the `BehaviorSubject` will not emit any items to
37  * subsequent subscribers, but instead it will pass along the error notification.
38  *
39  * 
40  */
41 final class BehaviorSubject[T](initialValue: T) extends Subject[T,T] { self =>
42   import monifu.reactive.subjects.BehaviorSubject.State
43   import monifu.reactive.subjects.BehaviorSubject.State._
44 
45   private[this] val state = Atomic(Empty(initialValue) : State[T])
46 
47   def subscribeFn(subscriber: Subscriber[T]): Unit = {
48     @tailrec
49     def loop(): ConnectableSubscriber[T] = {
50       state.get match {
51         case current @ Empty(cachedValue) =>
52           val obs = ConnectableSubscriber[T](subscriber)
53           obs.pushNext(cachedValue)
54 
55           if (!state.compareAndSet(current, Active(Array(obs), cachedValue)))
56             loop()
57           else
58             obs
59 
60         case current @ Active(observers, cachedValue) =>
61           val obs = ConnectableSubscriber[T](subscriber)
62           obs.pushNext(cachedValue)
63 
64           if (!state.compareAndSet(current, Active(observers :+ obs, cachedValue)))
65             loop()
66           else
67             obs
68 
69         case current @ Complete(cachedValue, errorThrown) =>
70           val obs = ConnectableSubscriber[T](subscriber)
71           if (errorThrown eq null) {
72             obs.pushNext(cachedValue)
73             obs.pushComplete()
74           }
75           else {
76             obs.pushError(errorThrown)
77           }
78           obs
79       }
80     }
81 
82     loop().connect()
83   }
84 
85   @tailrec
86   def onNext(elem: T): Future[Ack] = {
87     state.get match {
88       case current @ Empty(_) =>
89         if (!state.compareAndSet(current, Empty(elem)))
90           onNext(elem)
91         else
92           Continue
93 
94       case current @ Active(observers, cachedValue) =>
95         if (!state.compareAndSet(current, Active(observers, elem)))
96           onNext(elem)
97 
98         else
99           stream(observers, elem)
100 
101       case _ =>
102         Cancel
103     }
104   }
105 
106   @tailrec
107   def onComplete(): Unit =
108     state.get match {
109       case current @ Empty(cachedValue) =>
110         if (!state.compareAndSet(current, Complete(cachedValue, null))) {
111           onComplete() // retry
112         }
113       case current @ Active(observers, cachedValue) =>
114         if (!state.compareAndSet(current, Complete(cachedValue, null))) {
115           onComplete() // retry
116         }
117         else {
118           var idx = 0
119           while (idx < observers.length) {
120             observers(idx).observer.onComplete()
121             idx += 1
122           }
123         }
124       case _ =>
125         // already complete, ignore
126     }
127 
128   @tailrec
129   def onError(ex: Throwable): Unit =
130     state.get match {
131       case current @ Empty(cachedValue) =>
132         if (!state.compareAndSet(current, Complete(cachedValue, ex))) {
133           onError(ex) // retry
134         }
135       case current @ Active(observers, cachedValue) =>
136         if (!state.compareAndSet(current, Complete(cachedValue, ex))) {
137           onError(ex) // retry
138         }
139         else {
140           var idx = 0
141           while (idx < observers.length) {
142             observers(idx).observer.onError(ex)
143             idx += 1
144           }
145         }
146       case _ =>
147         // already complete, ignore
148     }
149 
150   private[this] def stream(array: Array[ConnectableSubscriber[T]], elem: T): Future[Continue] = {
151     val newPromise = PromiseCounter[Continue](Continue, array.length)
152     val length = array.length
153     var idx = 0
154 
155     while (idx < length) {
156       val subscriber = array(idx)
157       val obs = subscriber.observer
158       implicit val s = subscriber.scheduler
159 
160       obs.onNext(elem).onCompleteNow {
161         case Continue.IsSuccess =>
162           newPromise.countdown()
163         case _ =>
164           removeSubscription(subscriber)
165           newPromise.countdown()
166       }
167 
168       idx += 1
169     }
170 
171     newPromise.future
172   }
173 
174   @tailrec
175   private[this] def removeSubscription(subscriber: Subscriber[T]): Unit =
176     state.get match {
177       case current @ Active(subscribers, cachedValue) =>
178         val update = subscribers.filterNot(_ == subscriber)
179         if (update.nonEmpty) {
180           if (!state.compareAndSet(current, Active(update, cachedValue)))
181             removeSubscription(subscriber)
182         }
183         else {
184           if (!state.compareAndSet(current, Empty(cachedValue)))
185             removeSubscription(subscriber)
186         }
187       case _ =>
188         () // ignore
189     }
190 }
