17 package monifu.concurrent.cancelables
18 
19 import monifu.concurrent.atomic.Atomic
20 import monifu.concurrent.Cancelable
21 import scala.annotation.tailrec
22 
23 /**
24  * Represents a `Cancelable` that only executes the canceling logic when all
25  * dependent cancelable objects have been canceled.
26  *
27  * After all dependent cancelables have been canceled, `onCancel` gets called.
28  */
29 final class RefCountCancelable private (onCancel: () => Unit) extends BooleanCancelable {
30   def isCanceled: Boolean =
31     state.get.isCanceled
32 
33   @tailrec
34   def acquire(): Cancelable = {
35     val oldState = state.get
36     if (oldState.isCanceled)
37       BooleanCancelable.alreadyCanceled
38     else if (!state.compareAndSet(oldState, oldState.copy(activeCounter = oldState.activeCounter + 1)))
39       acquire()
40     else
41       Cancelable {
42         val newState = state.transformAndGet(s => s.copy(activeCounter = s.activeCounter - 1))
43         if (newState.activeCounter == 0 && newState.isCanceled)
44           onCancel()
45       }
46   }
47 
48   def cancel(): Boolean = {
49     val oldState = state.get
50     if (!oldState.isCanceled)
51       if (!state.compareAndSet(oldState, oldState.copy(isCanceled = true)))
52         cancel()
53       else if (oldState.activeCounter == 0) {
54         onCancel()
55         true
56       }
57       else
58         true
59     else
60       false
61   }
62 
63   private[this] val state = Atomic(State(isCanceled = false, activeCounter = 0))
64   private[this] case class State(
65     isCanceled: Boolean,
66     activeCounter: Int
67   )
68 }
69 
70 object RefCountCancelable {
71   def apply(onCancel: => Unit): RefCountCancelable =
72     new RefCountCancelable(() => onCancel)
73 }
