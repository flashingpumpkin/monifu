19 import monifu.concurrent.Cancelable
20 import monifu.concurrent.atomic.padded.Atomic

22 /**
23  * Represents a Cancelable that can queried for the canceled status.
24  */
25 trait BooleanCancelable extends Cancelable {
26   /**
27    * @return true in case this cancelable hasn't been canceled.
28    */
29   def isCanceled: Boolean
30 }
31 
32 object BooleanCancelable {
33   def apply(): BooleanCancelable =
34     new BooleanCancelable {
35       private[this] val _isCanceled = Atomic(false)
36       def isCanceled = _isCanceled.get
37 
38       def cancel(): Boolean = {
39         _isCanceled.compareAndSet(expect = false, update = true)
40       }
41     }
42 
43   def apply(callback: => Unit): BooleanCancelable =
44     new BooleanCancelable {
45       private[this] val _isCanceled = Atomic(false)
46 
47       def isCanceled =
48         _isCanceled.get
49 
50       def cancel(): Boolean = {
51         if (_isCanceled.compareAndSet(expect=false, update=true)) {
52           callback
53           true
54         }
55         else
56           false
57       }
58     }
59 
60   val alreadyCanceled: BooleanCancelable =
61     new BooleanCancelable {
62       val isCanceled = true
63       def cancel() = false
64     }
65 }
