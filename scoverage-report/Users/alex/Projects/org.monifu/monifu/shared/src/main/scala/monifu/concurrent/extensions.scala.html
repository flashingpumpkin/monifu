24 object extensions {
25   /**
26    * Provides utility methods added on Scala's `concurrent.Future`
27    */
28   implicit class FutureExtensions[T](val source: Future[T]) extends AnyVal {
29     /**
30      * Combinator that returns a new Future that either completes with
31      * the original Future's result or with a TimeoutException in case
32      * the maximum wait time was exceeded.
33      *
34      * @param atMost specifies the maximum wait time until the future is
35      *               terminated with a TimeoutException
36      * @param s is the Scheduler, needed for completing our internal promise
37      *
38      * @return a new future that will either complete with the result of our
39      *         source or fail in case the timeout is reached.
40      */
41     def withTimeout(atMost: FiniteDuration)(implicit s: Scheduler): Future[T] =
42       FutureUtils.withTimeout(source, atMost)
43 
44     /**
45      * Utility that lifts a `Future[T]` into a `Future[Try[T]]`, just because
46      * it is useful sometimes.
47      */
48     def liftTry(implicit ec: ExecutionContext): Future[Try[T]] =
49       FutureUtils.liftTry(source)
50 
51     /**
52      * Returns a new `Future` that takes a minimum amount of time to execute,
53      * specified by `atLeast`.
54      *
55      * @param atLeast the minimal duration that the returned future will take to complete.
56      * @param s the implicit scheduler that handles the scheduling and the execution
57      * @return a new `Future` whose execution time is within the specified bounds
58      */
59     def withMinDuration(atLeast: FiniteDuration)(implicit s: Scheduler): Future[T] =
60       FutureUtils.withMinDuration(source, atLeast)
61   }
62 
63   /**
64    * Provides utility methods for Scala's `concurrent.Future` companion object.
65    */
66   implicit class FutureCompanionExtensions(val f: Future.type) extends AnyVal {
67     /**
68      * Creates a future that completes with the specified `result`, but only
69      * after the specified `delay`.
70      */
71     def delayedResult[T](delay: FiniteDuration)(result: => T)(implicit s: Scheduler): Future[T] =
72       FutureUtils.delayedResult(delay)(result)
73   }
