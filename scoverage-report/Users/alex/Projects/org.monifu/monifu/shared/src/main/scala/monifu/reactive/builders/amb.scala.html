25 object amb {
26   /**
27    * Given a list of source Observables, emits all of the items from the first of
28    * these Observables to emit an item and cancel the rest.
29    */
30   def apply[T](source: Observable[T]*): Observable[T] = {
31     // helper function used for creating a subscription that uses `finishLine` as guard
32     def createSubscription(observable: Observable[T], observer: Observer[T], finishLine: AtomicInt, idx: Int)(implicit s: Scheduler): Unit =
33       observable.unsafeSubscribe(new Observer[T] {
34         def onNext(elem: T): Future[Ack] = {
35           if (finishLine.get == idx || finishLine.compareAndSet(0, idx))
36             observer.onNext(elem)
37           else
38             Cancel
39         }
40 
41         def onError(ex: Throwable): Unit = {
42           if (finishLine.get == idx || finishLine.compareAndSet(0, idx))
43             observer.onError(ex)
44         }
45 
46         def onComplete(): Unit = {
47           if (finishLine.get == idx || finishLine.compareAndSet(0, idx))
48             observer.onComplete()
49         }
50       })
51 
52     Observable.create { subscriber =>
53       implicit val s = subscriber.scheduler
54       val observer = subscriber.observer
55 
56       val finishLine = Atomic(0)
57       var idx = 0
58       for (observable <- source) {
59         createSubscription(observable, observer, finishLine, idx + 1)
60         idx += 1
61       }
62 
63       // if the list of observables was empty, just
64       // emit `onComplete`
65       if (idx == 0) observer.onComplete()
66     }
67   }
68 }
